```go
package main

import "fmt"

func main() {
  var a, b, c, d, e []byte
  fmt.Scan(&a, &b, &c, &d, &e)
  r := make([]byte, 0, len(a)+len(b)+len(c)+len(d)+len(e))
  for len(a)+len(b)+len(c)+len(d)+len(e) > 0 {
    if len(a) > 0 {
      r, a = append(r, a[0]), a[1:]
    }
    if len(b) > 0 {
      r, b = append(r, b[0]), b[1:]
    }
    if len(c) > 0 {
      r, c = append(r, c[0]), c[1:]
    }
    if len(d) > 0 {
      r, d = append(r, d[0]), d[1:]
    }
    if len(e) > 0 {
      r, e = append(r, e[0]), e[1:]
    }
  }
  fmt.Println(string(r))
}
```
