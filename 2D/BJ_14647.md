# [BJ_14647](https://acmicpc.net/problem/14647)

Get rid of row/column with most number of 9s written across all rows and columns and find number of remaining 9

```txt
Input:
3 4
1 2 3 9
4 5 9 6
9 7 8 9

Output: 2
```

## Solution

```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func next() []byte {
  in.Scan()
  return in.Bytes()
}

func main() {
  in.Split(bufio.ScanWords)
  n, m := nextInt(), nextInt()
  z, q, w := make([]int, n+m), 0, 0
  for i := 0; i < n; i++ {
    for j := 0; j < m; j++ {
      for _, c := range next() {
        if c == '9' {
          q++
          z[i]++
          z[j+n]++
        }
      }
    }
  }
  for _, v := range z {
    if w < v {
      w = v
    }
  }
  fmt.Println(q - w)
}
```

```cpp
#include<iostream>
using namespace std;
int max_val;
int arr[500][500];
int row[500], col[500];
int sum;

int func(int num) {
  int cnt = 0;
  while (num) {
    if (num % 10== 9) cnt++;
    num /= 10;
  }
  return cnt;
}

int main() {
  cin.tie(0);
  cin.sync_with_stdio(false);
  int n, m;
  cin >> n >> m;
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
      cin >> arr[i][j];
      row[i] += func(arr[i][j]);
      col[j] += func(arr[i][j]);
      sum += func(arr[i][j]);
    }
  }
  for (int i = 0; i < n; i++)
    if (row[i] > max_val) max_val = row[i];

  for (int j = 0; j < m; j++)
    if (col[j] > max_val) max_val = col[j];

  cout << sum - max_val <<endl;
}
```
