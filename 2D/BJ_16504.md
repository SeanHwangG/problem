```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int N; cin >> N;
  long long ans = 0;
  for (int i = 0; i < N; i++) {
    for (int j = 0; j < N; j++) {
      int x; cin >> x;
      ans += x;
    }
  }
  cout << ans << "\n";
}
```
