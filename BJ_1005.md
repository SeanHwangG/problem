import sys
input = sys.stdin.readline

def topologicalSort(G):
  indegree = [0] * N
  for dic in G:
    for adj in dic.keys():
      indegree[adj] += 1

  order, id2cost = [], [0] * N
  for i in range(N):
    if indegree[i] == 0:
      id2cost[i] = D[i]
      order.append(i)
  for v in order:
    for adj, cost in G[v].items():
      id2cost[adj] = max(id2cost[adj], id2cost[v] + cost)
      indegree[adj] -= 1
      if indegree[adj] == 0:
        order.append(adj)
  return id2cost

T = int(input())
for _ in range(T):
  N, K = map(int, input().split())
  D = list(map(int, input().split()))
  G = [{} for _ in range(N)]
  for _ in range(K):
    X, Y = map(int, input().split())
    G[X - 1][Y - 1] = D[Y - 1]

  W = int(input()) - 1
  print(topologicalSort(G)[W])
