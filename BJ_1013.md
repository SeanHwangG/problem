```go
package main

import (
  "bytes"
  "io/ioutil"
  "os"
  "regexp"
)

func main() {
  r := map[bool]string{false: "NO\n", true: "YES\n"}
  p := regexp.MustCompile("^((100+1+)|(01))+$")
  b, _ := ioutil.ReadAll(os.Stdin)
  for _, b := range bytes.Fields(b)[1:] {
    os.Stdout.WriteString(r[p.Match(b)])
  }
}
```
