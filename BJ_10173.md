package main

import (
  "bufio"
  "fmt"
  "os"
  "strings"
)

var in = bufio.NewScanner(os.Stdin)

func next() string {
  in.Scan()
  return in.Text()
}

func main() {
  for main2() {
  }
}

func main2() bool {
  s := next()
  if s == "EOI" {
    return false
  }
  if strings.Contains(strings.ToLower(s), "nemo") {
    fmt.Println("Found")
  } else {
    fmt.Println("Missing")
  }
  return true
}
