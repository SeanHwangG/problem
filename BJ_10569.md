package main

import "fmt"

func main() {
  var t int
  var v, e int
  fmt.Scanf("%d", &t)
  for i := 0; i < t; i++ {
    fmt.Scanf("%d %d", &v, &e)
    fmt.Printf("%d\n", 2-v+e)
  }

}
