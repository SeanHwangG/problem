```py
total, win = map(int, input().split())
z = int(100 * win / total)
if z >= 99:
  print(-1)
else:
  lo, hi = 0, 1000000000
  while lo < hi:
    mi = (lo + hi) // 2
    if z < 100 * (win + mi) // (total + mi):
      hi = mi
    else:
      lo = mi + 1
  print(lo)
```
