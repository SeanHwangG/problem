* There are:-) for happy faces and:-( for sad faces
* When you get a text message, find which occurs more

```txt
Input: How are you :-) doing :-( today :-)?
Output: happy
```

{% endtab %}{% tab title='BJ_10769.go' %}

```go
package main

import (
  "bytes"
  "io/ioutil"
  "os"
)

func main() {
  s, _ := ioutil.ReadAll(os.Stdin)
  x := bytes.Count(s, []byte(":-)"))
  y := bytes.Count(s, []byte(":-("))
  r := "unsure"
  switch {
  case x == 0 && y == 0:
    r = "none"
  case x > y:
    r = "happy"
  case x < y:
    r = "sad"
  }
  os.Stdout.WriteString(r)
}
```

{% endtab %}{% tab title='BJ_10769.py' %}

```py
s = input()
happy = s.count(":-)")
sad = s.count(":-(")
if happy < sad:
  print("sad")
elif happy > sad:
  print("happy")
elif happy == sad != 0:
  print("unsure")
else:
  print("none")
```
