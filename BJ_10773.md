```cpp
#include<iostream>
#include<stack>
using namespace std;

int main(){
  stack<int> st;
  int n, x;
  cin >> n;
  while(n--) {
    cin >> x;
    if(!x) st.pop();
    else st.push(x);
  }
  x = 0;
  while(st.size()){
    x += st.top();
    st.pop();
  }
  cout << x;
}
```
