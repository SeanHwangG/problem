```cpp
#include <iostream>
using namespace std;

int main() {
  int n, a, s = 0;
  cin >> n;
  for (int i = 0; i < 5; i++) {
    cin >> a;
    if(n == a) s++;
  }
  cout << s;
}
```

    }
  }

```java
import java.util.Scanner;

public class Main {
  public static void main(String[] args) {
    var sc = new Scanner(System.in);
    var t = sc.nextInt();
    var k = 0;
    for (int i = 0; i < 5; i++) {
      var n = sc.nextInt();
      if (n == t) {
        k++;
      }
    }
    System.out.println(k);
  }
}
```
