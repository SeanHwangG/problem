```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func init() {
  in.Split(bufio.ScanWords)
}

func NextInt() int {
  in.Scan()
  r := 0
  flag := 1
  for _, c := range in.Bytes() {
    if c == '-' {
      flag = -1
      continue
    }
    r *= 10
    r += int(c - '0')
  }
  return r * flag
}

func main() {
  m := make(map[int]int)
  for n := NextInt(); n > 0; n-- {
    m[NextInt()]++
  }
  fmt.Println(m[NextInt()])
}
```
