* Given a total of N integers, determine how many v are there

```txt
Input:
11
1 4 1 2 4 2 4 2 3 4 4
2

Output: 3
```

{% endtab %}{% tab title='BJ_10807.cpp' %}

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
  int n, v;
  cin >> n;
  vector<int> x(n);
  for (int i = 0; i < n; ++i)
    cin >> x[i];
  cin >> v;
  cout << count(x.begin(), x.end(), v);
  return 0;
}
```
