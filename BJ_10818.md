package main

import (
  "bufio"
  "fmt"
  "os"
  "sort"
)

func main() {
  var n int
  reader := bufio.NewReader(os.Stdin)

  fmt.Fscanln(reader, &n)
  arr := make([]int, n)

  for i := range arr {
    fmt.Fscanf(reader, "%d ", &arr[i])
  }

  nums := sort.IntSlice(arr)
  nums.Sort()
  fmt.Println(nums[0], nums[len(nums)-1])
}
