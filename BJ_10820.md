```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func main() {
  for in.Scan() {
    var a, b, c, d int
    for _, s := range in.Bytes() {
      switch {
      case 'a' <= s && s <= 'z':
        a++
      case 'A' <= s && s <= 'Z':
        b++
      case '0' <= s && s <= '9':
        c++
      case s == ' ':
        d++
      }
    }
    fmt.Println(a, b, c, d)
  }
}
```

```py
import sys
while True:
  line = sys.stdin.readline().rstrip('\n')
  up, lo, sp, nu = 0, 0, 0, 0
  if not line:
    break
  for l in line:
    if l.isupper():
      up += 1
    elif l.islower():
      lo += 1
    elif l.isdigit():
      nu += 1
    elif l.isspace():
      sp += 1
  print(f"{lo} {up} {nu} {sp}")
```
