from functools import lru_cache
N = int(input())
G = []
for _ in range(N):
  G.append(list(map(int,input().split())))

@lru_cache(None)
def find_path(cur, visited):
  if visited == (1 << N) - 1:
    return G[cur][0] or float('inf')

  mn = float('inf')
  for nxt in range(N):
    if visited & (1 << nxt) == 0 and G[cur][nxt] != 0:
      mn = min(mn, find_path(nxt, visited | (1 << nxt)) + G[cur][nxt])
  return mn

print(find_path(0, 1))
