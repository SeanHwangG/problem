* print previous permutation (-1 if first)

```txt
Input:
5
5 4 3 2 1

Output:
5 4 3 1 2
```

{% endtab %}{% tab title='BJ_10973.py' %}

```cpp
#include <bits/stdc++.h>

using namespace std;

int N, ar[11000];
int main() {
  cin >> N;
  for (int x = 0; x < N; x++) cin>>ar[x];
  if (prev_permutation(ar, ar + N)) {
    for (int x = 0; x < N; x++)
      cout << ar[x] << " ";
  }
  else cout << "-1";
}
```
