```py
from collections import deque
n, l = map(int,input().split())
li = [*map(int,input().split())]
d = deque()
for i in range(n):
  if d and d[0]<i-l+1:
    d.popleft()
  while d and li[d[-1]] >= li[i]:
    d.pop()
  d.append(i)
  print(li[d[0]])
```
