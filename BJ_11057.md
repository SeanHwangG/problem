```cpp
#include <stdio.h>
int D[1020][10];
int main(){
  int i,j,n;
  scanf("%d",&n);
  for(i=0;i<n+10;++i){
    D[i][0]=1;
    for(j=1;j<10&&j<=i;++j)D[i][j]=(D[i-1][j]+D[i-1][j-1])%10007;
  }
  printf("%d",D[n+9][9]);
  return 0;
}
```
