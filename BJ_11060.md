```py
input()
li = list(map(int, input().split()))
dp = [0] + [1000] * (len(li) - 1)
for i in range(len(li)):
  for step in range(1, li[i] + 1):
    if i + step < len(dp):
      dp[i + step] = min(dp[i + step], dp[i] + 1)

print(dp[-1] if dp[-1] < 1000 else -1)
```
