```py
n, k, p = map(int, input().split())
c = [[0] * p for i in range(p)]
for i in range(p):
  c[i][0] = 1
  for j in range(1,i+1):
    c[i][j] = (c[i-1][j-1] + c[i-1][j]) % p
ans = 1
while n:
  ans = (ans * c[n % p][k % p]) % p
  n //= p
  k //= p
print(ans)
```
