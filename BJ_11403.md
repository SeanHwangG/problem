```py
N = int(input())
G = [input().split() for _ in range(N)]

for i in range(N):
  for j in range(N):
    for k in range(N):
      if G[j][i] == '1' and G[i][k] == '1':
        G[j][k] = '1'

for i in range(N):
  print(' '.join(G[i]))
```
