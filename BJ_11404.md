```py
V, E = int(input()), int(input())
G = [[float('inf')] * V for _ in range(V)]
for _ in range(E):
  u, v, w = map(int, input().split())
  G[u - 1][v - 1] = min(w, G[u - 1][v - 1])

for i in range(V):
  for j in range(V):
    for k in range(V):
      if G[j][i] + G[i][k] < G[j][k]:
        G[j][k] = G[j][i] + G[i][k]

for i in range(V):
  for j, v in enumerate(G[i]):
    if i == j or v == float('inf'):
      print(0, end=' ')
    else:
      print(v, end=' ')
  print()
```
