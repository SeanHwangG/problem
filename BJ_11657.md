```py
INF = int(1e9)
read = lambda: map(int,input().split(" "))

n, m = read()
e = [tuple(read()) for _ in range(m)]
neg_cycle = True
d = [INF] * (n+1)
d[1] = 0

for _ in range(n):
  neg_cycle = False
  for x, y, w in e:
    if d[x] < INF and d[y] > d[x] + w:
      neg_cycle = True
      d[y] = d[x] + w
  if not neg_cycle: break

if neg_cycle:
  print(-1)
else:
  for i in range(2, n+1):
    print(d[i] if d[i] < INF else -1)
```
