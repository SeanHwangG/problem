#include <bits/stdc++.h>
using namespace std;
int solution(vector<int> v){
  int N = v.size(), ret = 0;
  vector<int> dp(N, -1);
  for (int n : v){
    int i = lower_bound(dp.begin(), dp.begin() + ret, n, [](int a, int b){return a > b;}) - dp.begin();
    dp[i] = n;
    if (ret == i){
      ret++;
    }
  }
  return ret;
}
int main() {
  int n;
  cin >> n;
  vector<int> v;
  while (n--){
    int buff;
    cin >> buff;
    v.push_back(buff);
  }
  cout << solution(v);
}
