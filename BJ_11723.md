```py
import sys
input = sys.stdin.readline
M = int(input())
my_list = [False] * 20  # 0 ~ 19
for _ in range(M):
  c = input().split()
  if len(c) > 1:
    n = int(c[1]) - 1  # 1 ~ 20 -> 0 ~ 19
  if c[0] == 'add':
    my_list[n] = True
  elif c[0] == 'remove':
    my_list[n] = False
  elif c[0] == 'check':
    print(1 if my_list[n] else 0)
  elif c[0] == 'toggle':
    my_list[n] = not my_list[n]
  elif c[0] == 'all':
    my_list = [True] * 20
  elif c[0] == 'empty':
    my_list = [False] * 20
```
