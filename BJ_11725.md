import sys
sys.setrecursionlimit(10**9)
def dfs(G, start, node2head):
  for adj in sorted(G[start]):
    if adj not in node2head:
      node2head[adj] = start
      dfs(G, adj, node2head)
N = int(input())
G = [[] for i in range(N+1)]
for _ in range(N - 1):
  a, b = map(int, input().split())
  G[a].append(b)
  G[b].append(a)
node2head = {1 : 0}
dfs(G, 1, node2head)
for i in range(2, N + 1):
  print(node2head[i])
