package main

import (
  "fmt"
  "math/bits"
)

func main() {
  var n uint
  fmt.Scan(&n)
  if bits.OnesCount(n) == 1 {
    fmt.Printf("1")
  } else {
    fmt.Printf("0")
  }
}
