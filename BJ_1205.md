```py
N, score, p = map(int, input().split())
if N == 0:
  print(1)
  exit()
li = list(map(int, input().split()))
if len(li) >= p and li[p - 1] >= score:
  print(-1)
else:
  li += [-1]
  for i in range(p):
    if li[i] <= score:
      print(i + 1)
      break
```
