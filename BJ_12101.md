n, k = map(int, input().split())
cnt = 0
def dfs(cur, total, string):
  global cnt
  if cur > total:
    return
  if cur == total:
    cnt += 1
    if cnt == k:
      print(string[1:])
      exit(0)
  dfs(cur + 1, total, string + '+1')
  dfs(cur + 2, total, string + '+2')
  dfs(cur + 3, total, string + '+3')

dfs(0, n, '')
print(-1)
