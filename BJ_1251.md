```py
import sys
s = sys.stdin.readline().rstrip()
arr = []
for i in range(0,len(s)-2):
  for j in range(i+1,len(s)-1):
    a = s[0:i+1][::-1]
    b = s[i+1:j+1][::-1]
    c = s[j+1:][::-1]
    arr.append(a+b+c)
arr.sort()
print(arr[0])
```
