```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}
func main() {
  in.Split(bufio.ScanWords)
  y, m := 0, 0
  for n := nextInt(); n > 0; n-- {
    v := nextInt()
    y += (v + 30) / 30 * 10
    m += (v + 60) / 60 * 15
  }
  switch {
  case y < m:
    fmt.Printf("Y %d", y)
  case m < y:
    fmt.Printf("M %d", m)
  default:
    fmt.Printf("Y M %d", y)
  }
}
```
