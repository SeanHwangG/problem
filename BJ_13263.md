def cross(x, y):
  ax, bx, ay, by = *hull[x], *hull[y]
  return (by - bx) / (ax - ay)

def insert(A, B):
  hull.append((A, B))
  while len(hull) > 2 and cross(-2, -3) > cross(-1, -2):
    hull.pop(-2)

n = int(input())
A = list(map(int,input().split()))
B = list(map(int,input().split()))
p = 0
dp = [0]
hull = [(B[0], 0)]
for i in range(1, n):
  while p + 1 < len(hull) and cross(p, p + 1) <= A[i]:
    p += 1
  dp.append(hull[p][0] * A[i] + hull[p][1])
  insert(B[i], dp[i])
