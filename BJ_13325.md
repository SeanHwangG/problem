#include<bits/stdc++.h>
using namespace std;
int k, a[1 << 21], r, i, t;
int main() {
  cin >> k;
  for (i = 2; i < 1 << k + 1; i++) scanf("%d", a + i);
  while (i -= 2) {
    r += t = max(a[i], a[i + 1]);
    a[i / 2] += t;
  }
  cout << r + t;
  return 0;
}
