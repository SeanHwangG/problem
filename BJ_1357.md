package main

import "fmt"

func rev(a int) int {
  if a < 10 {
    return a
  }
  if a < 100 {
    return a/10 + a%10*10
  }
  if a < 1000 {
    return a/100 + a/10%10*10 + a%10*100
  }
  return a/1000 + a/100%10*10 + a/10%10*100 + a%10*1000
}

func main() {
  x, y := 0, 0
  fmt.Scan(&x, &y)
  fmt.Println(rev(rev(x) + rev(y)))
}
