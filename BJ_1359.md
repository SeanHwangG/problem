```py
from itertools import combinations

N, M, K = map(int, input().split())
comb = combinations(range(1, N + 1), M)
comb2 = combinations(range(1, N + 1), M)
length = sum(1 for _ in combinations(range(1, N + 1), M))

res = 0
for elem in comb:
  for elem2 in comb2:
    if len(set(elem) & set(elem2)) >= K:
      res += 1
print(res / length)
```
