import (
  "bufio"
  "fmt"
  "math/big"
  "os"

var in = bufio.NewReader(os.Stdin)

func main() {
  r := new(big.Int)
  fmt.Fscanf(in, "%b", r)
  fmt.Printf("%o\n", r)
}
