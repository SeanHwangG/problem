```py
N=int(input())
L = []
for i in range(N):
  due, score = map(int,input().split())
  L.append((score, due))
L.sort(reverse=True)
busy, total = set(), 0
for i in range(N):
  score, due = L[i]
  for j in range(due, 0, -1):
    if j not in busy:
      busy.add(j)
      total += score
      break
print(total)
```
