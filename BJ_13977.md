from sys import stdin
input = stdin.readline

MOD = 1000000007
fac = [1]
for i in range(1,4000001):
  fac.append((fac[-1]*i)%MOD)
for i in range(int(input())):
  n, k = map(int,input().split())
  f = fac[n]
  f = (f * pow(fac[k], MOD-2, MOD)) % MOD
  f = (f * pow(fac[n-k], MOD-2, MOD)) % MOD
  print(f)
