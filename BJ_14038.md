* 'W' is win 'L' is lose
* print -1 if won less than 1
* print 3 if won less than 3
* print 2 if won less than 5
* print 1 if won greater or equal than 5

```txt
Input:
W
L
W
W
L
W

Output: 2
```

{% endtab %}{% tab title='BJ_14038.go' %}

```go
package main
import "fmt"

func main() {
  var s string
  var cnt int = 0
  for i := 0; i < 6; i++ {
    fmt.Scanf("%s", &s)
    if s == "W" {
      cnt += 1
    }
  }
  var res int
  if cnt >= 5 {
    res = 1
  } else if cnt >= 3 {
    res = 2
  } else if cnt >= 1 {
    res = 3
  } else {
    res = -1
  }
  fmt.Printf("%d\n", res)
}

```
