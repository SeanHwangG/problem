```go
package main

import (
  "fmt"
)

func main() {
  computer := make([]bool, 100)

  var n int
  fmt.Scan(&n)

  var count int

  for i := 0; i < n; i++ {
    var temp int
    fmt.Scan(&temp)

    if computer[temp-1] == true {
      count++
    } else {
      computer[temp-1] = true
    }
  }

  fmt.Print(count)
}
```
