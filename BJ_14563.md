```py
input()
for s in map(int, input().split()):
  tmp=[]
  for i in range(1, (s+1) // 2):
    if s % i == 0:
      if s // i == i : tmp.append(i)
      else: tmp.append(i);tmp.append(s//i)
  ans = sum(sorted(set(tmp))) - s
  if ans > s:print("Abundant")
  elif ans == s:print("Perfect")
  else: print("Deficient")
```
