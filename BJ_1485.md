def dist(p1, p2):
  ax, ay = p1
  bx, by = p2
  return (ax - bx) ** 2 + (ay - by) ** 2
for _ in range(int(input())):
  P = [tuple(map(int,input().split())) for i in range(4)]
  P.sort()
  ans = dist(P[0], P[3]) == dist(P[1], P[2]) and dist(P[0], P[1]) == dist(P[0], P[2])
  print(int(ans))
