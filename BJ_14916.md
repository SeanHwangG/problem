```go
package main

import "fmt"

func main() {
  n := 0
  fmt.Scan(&n)
  if n == 1 || n == 3 {
    fmt.Println("-1")
    return
  }
  r := n / 5
  switch n % 5 {
  case 1, 4:
    r += 2
  case 2:
    r++
  case 3:
    r += 3
  }
  fmt.Println(r)
}
```

```py
N = int(input())
for i in range(0, N + 1, 2):
  if (N - i) % 5 == 0:
    print(i // 2 + (N - i) // 5)
    break
else:
  print(-1)
```
