```py
n, m, r = map(int, input().split())
item = list(map(int, input().split()))
inf = 10**9
A = [[inf] * n for _ in range(n)]
for _ in range(r):
  a, b, c = map(int, input().split())
  A[a - 1][b - 1] = A[b - 1][a - 1] = c

ans = 0

for k in range(n):
  A[k][k] = 0
  for i in range(n):
    for j in range(n):
      A[i][j] = min(A[i][j], A[i][k] + A[k][j])

for y in range(n):
  s = 0
  for x in range(n):
    if A[y][x] <= m:
      s += item[x]
  ans = max(ans, s)

print(ans)
```
