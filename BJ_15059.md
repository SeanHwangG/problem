```go
package main
import "fmt"
func main() {
  var c1, b1, p1 int
  var c2, b2, p2 int
  fmt.Scanf("%d %d %d\n%d %d %d", &c1, &b1, &p1, &c2, &b2, &p2)
  var sum int = 0
  if c1 < c2 {
    sum += (c2 - c1)
  }
  if b1 < b2 {
    sum += (b2 - b1)
  }
  if p1 < p2 {
    sum += (p2 - p1)
  }
  fmt.Printf("%d\n", sum)
}
```
