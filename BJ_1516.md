```py
N, costs = int(input()), []
G, G_r  = [[] for i in range(N)], [[] for i in range(N)]
wait = [0] * N
for u in range(N):
  li = list(map(int, input().split()))
  costs.append(li[0])
  for v in li[1:-1]:
    G[v - 1].append(u)
    G_r[u].append(v - 1)
    wait[u] += 1

bfs = [n for n in range(N) if wait[n] == 0]
for i in bfs:
  for j in G[i]:
    wait[j] -= 1
    if wait[j] == 0:
      bfs.append(j)
for i in bfs:
  costs[i] += max([0] + [costs[j] for j in G_r[i]])
print(*costs, sep='\n')
```
