```py
from collections import deque
p = "".join(input().replace(" ", "") for _ in range(3))

visited = {p: 0}
qu = deque([p])

while qu:
  p = qu.popleft()
  if p == "123456780":
    print(visited[p])
    break

  a = p.find("0")
  r, c = divmod(a, 3)
  for nr, nc in (r + 1, c), (r, c + 1), (r - 1, c), (r, c - 1):
    try:
      b = nr * 3 + nc
      np = list(p)
      np[a], np[b] = np[b], np[a]
      np = "".join(np)
      if np not in visited:
        visited[np] = visited[p] + 1
        qu.append(np)
    except:
      pass  # out of bound
else:
  print(-1)
```
