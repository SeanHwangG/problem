```cpp
#include <stdio.h>
int n, m, vis[77];
void f(int x) {
  if (x > m) {
    for (int i = 1; i <= m; i++)
      printf("%d ", vis[i]);
    printf("\n");
    return;
  }
  for (int i = 1; i <= n; i++) {
    vis[x] = i;
    f(x + 1);
  }
}

int main()
{
  cin >> n >> m;

  f(1);
}
```
