N, M = map(int, input().split())
li = list(sorted(map(int, input().split())))
def backtrack(cur, N, M):
  if len(cur) == M:
    print(*cur)
    return
  for i in range(N):
    cur.append(li[i])
    backtrack(cur, N, M)
    cur.pop()
backtrack([], N, M)
