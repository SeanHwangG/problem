```py
N, i, j = map(int, input().split())
i, j = i - 1, j - 1
line = []
row = []
for a in range(N):
  temp = list(map(int, input().split()))
  line.append(temp[j])
  if a == i:
    row = temp
if max(line) == max(row) == row[j]:
  print('HAPPY')
else:
  print('ANGRY')
```
