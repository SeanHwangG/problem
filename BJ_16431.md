package main

import "fmt"

func abs(x int) int {
  if x < 0 {
    return -x
  }
  return x
}

func max(a, b int) int {
  if a > b {
    return a
  }
  return b
}

func main() {
  var br, bc, dr, dc, jr, jc int
  fmt.Scan(&br, &bc, &dr, &dc, &jr, &jc)
  b := max(abs(br-jr), abs(bc-jc))
  d := abs(dr-jr) + abs(dc-jc)
  switch {
  case b < d:
    fmt.Printf("bessie")
  case b > d:
    fmt.Printf("daisy")
  default:
    fmt.Printf("tie")
  }
}
