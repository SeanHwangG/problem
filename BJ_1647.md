def find(x):
  if P[x] != x:
    P[x] = find(P[x])
  return P[x]


def union(a, b):
  a, b = find(a), find(b)
  if a != b:
    P[b] = a
    return 0
  return 1


N, M = map(int, input().split())
Q, cnt, R = [list(map(int, input().split())) for _ in range(M)], 0, 0
Q.sort(key=lambda t: t[2])
P = list(range(N))
for m in Q:
  if cnt == N - 2:
    break
  a, b, c = m
  if not union(a - 1, b - 1):
    cnt += 1
    R += c
print(R)
