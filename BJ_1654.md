```py
n, pieces = map(int, input().split())
li = []
for _ in range(n):
  li.append(int(input()))
def count(li, length):
  ret = 0
  for n in li:
    ret += n // length
  return ret
def binary_search(li, x):
  lo, hi = 0, max(li)
  while lo < hi:
    mi = (hi + lo + 1) // 2
    if count(li, mi) < x:
      hi = mi - 1
    else:
      lo = mi
  return lo
print(binary_search(li, pieces))
```
