* Factorize given N integers

```txt
Input:
5
5 4 45 64 54

Output:
5
2 2
3 3 5
2 2 2 2 2 2
2 3 3 3
```

{% endtab %}{% tab title='BJ_16563.py' %}

```py
import sys
input = sys.stdin.readline

input()
arr = list(map(int, input().split()))
MAX = 5000001
least = [0] * MAX
for i in range(2, 2237):
  if not least[i]:
    for j in range(i*i, MAX, i):
      if not least[j]:
        least[j] = i

def divide(x):
  while least[x]:
    print(least[x], end=' ')
    x = x//least[x]
  print(x)

for i in arr:
  divide(i)
```
