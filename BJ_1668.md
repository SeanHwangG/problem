def count_visible(li):
  count, cur_mx = 0, 0
  for e in li:
    if e > cur_mx:
      count += 1
      cur_mx = e
  return count
li = [int(input()) for _ in range(int(input()))]
print(count_visible(li))
print(count_visible(reversed(li)))
