package main

import (
  "bufio"
  "os"
)

var in = bufio.NewScanner(os.Stdin)
var data = map[byte]map[byte]byte{
  'A': {'A': 'A', 'G': 'C', 'C': 'A', 'T': 'G'},
  'G': {'A': 'C', 'G': 'G', 'C': 'T', 'T': 'A'},
  'C': {'A': 'A', 'G': 'T', 'C': 'C', 'T': 'G'},
  'T': {'A': 'G', 'G': 'A', 'C': 'G', 'T': 'T'},
}

func next() []byte {
  in.Scan()
  return in.Bytes()
}

func main() {
  in.Buffer(make([]byte, 1048576), 1048576)
  _, b := next(), next()
  for i := len(b) - 2; i >= 0; i-- {
    b[i] = data[b[i]][b[i+1]]
  }
  os.Stdout.Write(b[:1])
}
