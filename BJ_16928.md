l, visited = [*range(102)], [0] * 101
for i in range(sum(map(int,input().split()))):
  x, y = map(int,input().split())
  l[x] = y
q, c = [1], 1
while 1:
  new = []
  for i in q:
    if i > 93: exit(print(c))
    for j in range(i + 1, i + 7):
      if not visited[l[j]]:
        visited[l[j]] = 1
        new += [l[j]]
  q, c = new, c + 1
