from collections import deque
def bfs(start, end):
  dq, node2dist = deque([start]), {start : 0}
  while len(dq) != 0:
    v = dq.popleft()
    if v == end:
      return node2dist[end]
    for adj in [v + 1, v - 1, v * 2]:
      if adj not in node2dist and adj <= 100000:
        node2dist[adj] = node2dist[v] + 1
        dq.append(adj)
start, end = map(int, input().split())
print(bfs(start, end))
