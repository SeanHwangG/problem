from functools import lru_cache
N = int(input())
P = [tuple(map(int, input().split())) for _ in range(N)]
G = [[((P[i][0] - P[j][0]) ** 2 + (P[i][1] - P[j][1]) ** 2) ** 0.5 for i in range(N)] for j in range(N)]
@lru_cache(None)
def dfs(cur, visited):
  if visited == (1 << N) - 1:
    return G[cur][0]
  mn = float("inf")
  for nxt in range(N):
    if visited & 1 << nxt == 0:
      mn = min(mn, dfs(nxt, visited | (1 << nxt)) + G[cur][nxt])
  return mn
print(dfs(0, 1))
