```py
from math import prod
from itertools import combinations

m, n = map(int, input().split())
p_set = list(map(int, input().split()))
ans = 0

for r in range(1, m+1):
  t = sum(n // prod(c) for c in combinations(p_set, r))
  ans += t if r & 1 else -t

print(ans)
```
