from itertools import combinations
n, m = map(int, input().split())
li = sorted(input().split())
for c in combinations(li, n):
  count = 0
  for letter in c:
    if letter in "aeiou":
      count += 1
  if (count >= 1) and (count <= n-2):
    print(*c, sep ='')
