```py
import sys
input=sys.stdin.readline
N = int(input())
G = [[]for i in range(N+1)]
for i in range(N-1):
  a, b, w = map(int,input().split())
  G[a].append([b, w])
  G[b].append([a, w])
D, P = [0 for i in range(N + 1)], [[0, 0] for i in range(N+1)]
D[1] = 1
st = [1]
while st:
  i=st.pop()
  for j, w in G[i]:
    if not D[j]:
      D[j] = D[i] + 1
      P[j] = [i, w]
      st.append(j)
M = int(input())
for _ in range(M):
  a, b = map(int, input().split())
  d = 0
  while D[a] != D[b]:
    if D[a] > D[b]:
      d += P[a][1]
      a = P[a][0]
    else:
      d += P[b][1]
      b = P[b][0]
  while a != b:
    d += P[a][1] + P[b][1]
    a, b = P[a][0], P[b][0]
  print(d)
```
