import sys
sys.setrecursionlimit(10**6)
input = sys.stdin.readline
def same(x, y, n):
  for i in range(x, x+n):
    for j in range(y, y+n):
      if G[x][y] != G[i][j]:
        return False
  return True
def solve(x, y, n):
  if same(x, y, n):
    cnt[G[x][y]+1] += 1
    return
  for i in range(0, 3):
    for j in range(0, 3):
      solve(x + i * n // 3, y + j * n // 3, n // 3)
cnt = [0] * 3
n = int(input())
G = [list(map(int, input().split())) for _ in range(n)]
solve(0, 0, n)
for i in cnt:
  print(i)
