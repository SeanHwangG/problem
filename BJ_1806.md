```py
N, s = map(int, input().split())
li = list(map(int, input().split()))
acc = [0]
for n in li:
  acc.append(acc[-1] + n)

mn_len = N
for i in range(N):
  lo, hi = i, N - 1
  if acc[hi + 1] - acc[i] < s:
    break
  while lo < hi:
    mi = (lo + hi) // 2
    if s <= acc[mi + 1] - acc[i]:
      hi = mi
    else:
      lo = mi + 1
  mn_len = min(mn_len, lo - i + 1)
print(0 if i == 0 else mn_len)
```
