```py
import sys
input = sys.stdin.readline
n, m, b = map(int, input().split())
G = [list(map(int, input().split())) for _ in range(n)]
height, ans = 0, float('inf')
for i in range(257):
  mx, mn = 0, 0
  for row in G:
    for e in row:
      if e < i:
        mn += i - e
      else:
        mx += e - i
  if mx + b >= mn:
    time = 2 * mx + mn
    if time <= ans:
      ans = time
      height = i
print(ans, height)
```
