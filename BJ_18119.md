```py
import sys
input = sys.stdin.readline
n, m = map(int, input().split())
words = [0 for i in range(n)]
for i in range(n):
  for ch in input().rstrip():
    words[i] |= 1 << (ord(ch) - 97)
alpha = 0xffffffff
for _ in range(m):
  o, x = input().split()
  if o == '1':
    alpha &= ~(1 << ord(x)-97)
  elif o == '2':
    alpha |= (1 << ord(x)-97)
  print(sum(1 for word in words if (word & alpha) == word))
```
