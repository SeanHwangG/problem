```go
package main

import "fmt"

func main() {
  var num string
  code := map[rune]int{
    '-':  0,
    '\\': 1,
    '(':  2,
    '@':  3,
    '?':  4,
    '>':  5,
    '&':  6,
    '%':  7,
    '/':  -1,
  }
  for {
    fmt.Scan(&num)
    if num == "#" {
      break
    }
    res := 0
    for i, c := range num {
      res += code[c] << (3*(len(num)-1-i))
    }
    fmt.Println(res)
  }
}
```

```py
ch2int = {'-':0, '\\':1, '(':2, '@':3, '?':4, '>':5, '&':6, '%':7, '/':-1}
while True:
  string = input()
  if string == '#':
    break
  ret = 0
  for ch in string:
    ret *= 8
    ret += ch2int[ch]
  print(ret)
```
