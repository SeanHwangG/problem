s = [0]
x, v, r = 0, 0, ''
for i in [int(input()) for _ in range(int(input()))]:
  if s[-1] != i:
    v += x > i
    while x<i:
      x += 1
      s += [x]
      r += '+'
  r += '-'
  s.pop()
print('NO'if v else'\n'.join(r))
