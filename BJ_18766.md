#include <bits/stdc++.h>
using namespace std;
int main() {
  int T; cin >> T;
  while(T--) {
    int N; cin >> N;
    vector<string> v(N), v2(N);
    for (auto &i : v)cin >> i;
    for (auto &i : v2)cin >> i;
    sort(v.begin(), v.end());
    sort(v2.begin(), v2.end());
    if (v == v2) cout << "NOT CHEATER\n";
    else cout << "CHEATER\n";
  }
}
