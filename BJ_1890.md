```py
n = int(input())
G = [list(map(int, input().split())) for _ in range(n)]
dp = [[0 for _ in range(n)] for _ in range(n)]
dp[0][0] = 1
for y in range(n):
  for x in range(n):
    if dp[y][x] != 0 and G[y][x] != 0:
      if y + G[y][x] < n:
        dp[y + G[y][x]][x] += dp[y][x]
      if x + G[y][x] < n:
        dp[y][x+ G[y][x]] += dp[y][x]
print(dp[-1][-1])
```
