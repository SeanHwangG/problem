import sys
from heapq import heappush, heappop

input = sys.stdin.readline

v, e = int(input()), int(input())
G = [[] for _ in range(v + 5)]
for _ in range(e):
  u, v, w = map(int, input().split())
  G[u].append((w, v))
s, e = map(int, input().split())
pq = [(0, s)]
dist = [0 if i == s else 1 << 30 for i in range(v + 5)]
while len(pq):
  w, u = heappop(pq)
  if w > dist[u]:
    continue
  for d, v in G[u]:
    if dist[v] > d + w:
      dist[v] = d + w
      heappush(pq, (d + w, v))
print(dist[e])
