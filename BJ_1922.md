```py
import heapq
n,  m = int(input()), int(input())
G = [[] for _ in range(n)]
for _ in range(m):
  s, e, c = map(int, input().split())
  G[s - 1].append([e - 1, c])
  G[e - 1].append([s - 1, c])
key = [0] + [float('inf')] * (n - 1)
visited = [False] * n
pq = []
heapq.heappush(pq, (0, 0))
result = 0
while pq:
  k, u = heapq.heappop(pq)
  if visited[u]:
    continue
  visited[u] = True
  result += k
  for dest, w in G[u]:
    if not visited[dest] and w < key[dest]:
      key[dest] = w
      heapq.heappush(pq, (key[dest], dest))
print(result)
```
