```py
import heapq
import sys

input = sys.stdin.readline
heap = []

for _ in range(int(input())):
  num = int(input())
  if num != 0:
    heapq.heappush(heap, num)
  else:
    if len(heap):
      print(heapq.heappop(heap))
    else:
      print(0)
```
