```py
import sys
input = sys.stdin.readline
N = int(input())
discuss = [list(map(int, input().split())) for _ in range(N)]
cnt, time_now = 0, -1
for diss in sorted(discuss, key=lambda x: (x[1], x[0])):
  if diss[0] >= time_now: # Beginning of meeting
    time_now = diss[1]
    cnt+=1

print(cnt)
```
