```py
import sys
input = sys.stdin.readline
for _ in range(int(input())):
  n = int(input())
  score = sorted([tuple(map(int, input().split())) for _ in range(n)])
  p, ans = n + 1, 0
  for s, e in score:
    if p > e:
      ans += 1
      p = e
  print(ans)
```
