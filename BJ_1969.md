```py
n, m = map(int, input().split())
dna = [input().strip() for _ in range(n)]
dna_to_num = {'A': 0, 'C': 1, 'G': 2, 'T': 3}
res = ''
cnt = 0
for d in range(m):
  check = [0] * 4
  for i in range(n):
    check[dna_to_num[dna[i][d]]] += 1
  max_dna = max(check)
  max_dna_idx = check.index(max_dna)
  res += "ACGT"[max_dna_idx]
  check[max_dna_idx] = 0
  cnt += sum(check)
print(res)
print(cnt)
```
