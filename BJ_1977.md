```sh
read a
read b
c=0
d=100
s=0
while [ $((c*c)) -lt $a ]; do
  c=$((c+1))
done
while [ $((d*d)) -gt $b ]; do
  d=$((d-1))
done
e=$c
until [ $c -gt $d ]; do
  s=$((c*c+s))
  c=$((c+1))
done
if [ $s -eq 0 ]; then
  echo -1
else
  echo -e "$s\n$((e*e))"
fi
```
