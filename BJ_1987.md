def BFS(G, x, y):
  answer = 1
  q = set([(x, y, G[x][y])])
  while q:
    x, y, ans = q.pop()
    for nr, nc in [(x + 1, y), (x, y + 1), (x - 1, y), (x, y - 1)]:
      if 0 <= nr < len(G) and 0 <= nc < len(G[0]) and G[nr][nc] not in ans:
        q.add((nr,nc,ans + G[nr][nc]))
        answer = max(answer, len(ans) + 1)
  return answer
R, C = map(int, input().split())
G = [input() for _ in range(R)]
