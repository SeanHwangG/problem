```py
N, M = map(int, input().split())
nums = list(map(int, input().split()))
accums = [0] * (N + 1)
for i in range(1, N + 1):
  accums[i] = accums[i - 1] + nums[i - 1]
count = 0
for l in range(N):
  for r in range(l + 1, N + 1):
    if accums[r] - accums[l] > M:
      break
    elif accums[r] - accums[l] == M:
      count += 1
      break
print(count)
```
