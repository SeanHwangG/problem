```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int N; cin >> N;
  while (N--) {
    int n; cin >> n;
    vector<int> v(n);
    for (auto& i : v) cin >> i;
    cout << *min_element(v.begin(), v.end()) << ' ' << *max_element(v.begin(), v.end()) << '\n';
  }
}
```
