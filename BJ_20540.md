* Print opposite MBTI
  * E, I
  * S, N
  * T, F
  * J, P

```txt
Input: ESTJ
Output: INFP

Input: INFP
Output: ESTJ
```

{% endtab %}{% tab title='BJ_20540.asm' %}

```asm
section .bss
buf: resb 4
buf_len: equ $-buf

section .text
global main
main:
  xor eax, eax
  xor edi, edi
  lea rsi, [rel buf]
  mov rdx, buf_len
  syscall

  sub dword [rsi], 0x9A9AA18E
  neg dword [rsi]

  mov rax, 1
  mov rdi, 1
  syscall

  mov rax, 231
  xor edi, edi
  syscall
```
