```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var out = bufio.NewWriter(os.Stdout)

func main() {
  var n int
  fmt.Scan(&n)
  a := make([]int, n, n*2)
  for i := range a {
    a[i] = i + 1
  }
  for len(a) > 1 {
    fmt.Fprintf(out, "%d ", a[0])
    a = append(a, a[1])
    a = a[2:]
  }
  fmt.Fprint(out, a[0])
  out.Flush()
}
```

const N = +require('fs').readFileSync('dev/stdin').toString();
const arr = Array.apply(null, {length: N+1}).map(Number.call, Number);
arr.shift();

let output = "";
while (arr[0]) {
  output += arr.shift() + ' ';
  arr.push(arr.shift());
}
console.log(output);
