```cpp
#include<iostream>
#include<queue>
using namespace std;

int N;
queue<int> q;

int main() {
  cin >> N;
  for (int i = 1; i <= N; i++) q.push(i);
  while (q.size() != 1) {
    q.pop();
    q.push(q.front());
    q.pop();
  }
  cout << q.front();
}
```

```py
from collections import deque
deck = deque(range(1, int(input()) + 1))
while len(deck) > 1:
  deck.popleft()
  deck.append(deck.popleft())
```
