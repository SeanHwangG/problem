```py
N, M = map(int, input().split())
G = [list(map(int, input().split())) for _ in range(N)]
S, dp = 0, []
for i in range(M):
  S += G[0][i]
  dp.append(S)

for j in range(1, N):
  L = [dp[0] + G[j][0]] + [0] * (M - 1)
  for i in range(1, M):
    L[i] = G[j][i] + max(L[i - 1], dp[i])
  R = [0] * (M - 1) + dp[-1] + G[j][-1]
  for i in range(1, M):
    R[-i-1] = G[j][-i-1] + max(R[-i], dp[-i-1])
  for i in range(M):
    dp[i] = max(L[i], R[i])
print(dp[-1])
```
