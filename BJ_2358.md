```py
from collections import Counter
N = int(input())
x_cnt = Counter()
y_cnt = Counter()
for _ in range(N):
  x, y = map(int, input().split())
  x_cnt[x] += 1
  y_cnt[y] += 1

ret = 0
for x in x_cnt:
  if x_cnt[x] > 1:
    ret += 1
for y in y_cnt:
  if y_cnt[y] > 1:
    ret += 1
print(ret)
```
