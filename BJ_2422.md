```py
N, M = map(int, input().split())
no_mat_combi = []
for _ in range(M):
  no_mat_combi.append(list(map(int, input().split())))
no_mat = set()
for item in no_mat_combi:
  for i in range(1, N + 1):
    if i not in item:
      no_mat.add(tuple(sorted(item + [i])))
answer = int(N * (N - 1) * (N - 2) / 6 - len(no_mat))
print(answer)
```
