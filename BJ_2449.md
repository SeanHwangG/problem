```py
n,k = map(int ,input().split())
li = [*map(int, input().split())]
dp = [[0 for i in range(n)] for j in range(n)]

for i in range(n):
  for j in range(i-1, -1, -1):
    dp[j][i] = 1<<30
    for k in range(j, i):
      dp[j][i] = min(dp[j][i], dp[j][k] + dp[k + 1][i] + (li[j] != li[k + 1]))
print(dp[0][-1])
```
