```py
n, m = map(int, input().split())
G = [[0] * n for _ in range(n)]
for _ in range(m):
  u, v = map(int, input().split())
  G[u - 1][v - 1] = 1
  G[v - 1][u - 1] = -1
for k in range(n):
  for i in range(n):
    for j in range(n):
      G[i][j] = G[k][j] if G[i][k] == G[k][j] != 0 else G[i][j]
print(sum(1 for li in G if li.count(0) == 1))
```
