* If all three numbers are same print (sum of all) * 10000 + 1000
* else if only two of them are same print (sum of two) * 100 + 1000
* else print max * 100

```txt
Input: 3 3 6
Output: 1300
```

{% endtab %}{% tab title='BJ_2480.go' %}

```go
package main

import "fmt"

func main() {
  var a, b, c, r int
  fmt.Scan(&a, &b, &c)
  switch {
  case a == b && a == c:
    r = 10000 + a * 1000
  case a == b || a == c:
    r = 1000 + a * 100
  case b == c:
    r = 1000 + b * 100
  case a > b && a > c:
    r = 100 * a
  case b > a && b > c:
    r = 100 * b
  case c > a && c > b:
    r = 100 * c
  }
  fmt.Println(r)
}
```

```py
a, b, c = map(int, input().split())
if a == b == c:
  print(10000 + a * 1000)
elif a == b or a == c:
  print(1000 + a * 100)
elif b == c:
  print(1000 + c * 100)
else:
  print(max(a, b, c) * 100)
```
