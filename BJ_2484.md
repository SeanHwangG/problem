m = 0
for _ in range(int(input())):
  a, b, c, d = sorted(map(int, input().split()))
  if a==d: s=50000+a*5000
  elif a==c or b==d: s = 10000 + b * 1000
  elif a==b and c==d: s = 2000 + a * 500 + c * 500
  elif a==b or b==c: s = 1000 + b * 100
  elif c==d: s = 1000 + c * 100
  else: s = d * 100
  if m < s: m=s
print(m)
