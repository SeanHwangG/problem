from math import lcm
n = int(input())
L, v = list(map(int,input().split())), [False]*n
r = 1
for i in range(n):
  if v[i]: continue
  v[i] = True
  j, c = i, 0
  while j != i or c == 0:
    c += 1
    j = L[j]-1
    v[j] = True
  r = lcm(r,c)
print(r)
