```py
import sys
n = int(sys.stdin.readline().strip())
seq = list(map(int,sys.stdin.readline().split()))

result = 1
increase = 1
decrease = 1
for i in range(1,n):
  if seq[i - 1] <= seq[i]:
    increase += 1
    result = min(result, increase)
  else:
    increase = 1
for i in range(1, n):
  if seq[i - 1] >= seq[i]:
    decrease += 1
    result = min(result, decrease)
  else:
    decrease = 1
print(result)
```
