G = []
for _ in range(9):
  G.append(list(map(int, input().split())))
mx, mr, mc = 0, 0, 0
for r in range(9):
  for c in range(9):
    if mx < G[r][c]:
      mx = G[r][c]
      mr, mc = r + 1, c + 1
print(mx)
print(mr, mc)
