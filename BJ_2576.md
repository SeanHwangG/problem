* Find odd's sum and minimum

```txt
Input:
12
77
38
41
53
92
85

Output:
256
41
```

{% endtab %}{% tab title='BJ_2576.go' %}

```go
package main

import "fmt"

func main() {
  sum := 0
  min := 100
  var num int
  for i := 0; i < 7; i++ {
    fmt.Scanln(&num)
    if (num & 1) == 0 {
      continue
    }
    sum += num
    if num < min {
      min = num
    }
  }
  if sum == 0 {
    fmt.Println("-1")
  } else {
    fmt.Println(sum)
    fmt.Println(min)
  }
}
```

```py
sm = 0
mn = 100
for _ in range(7):
  a = int(input())
  if a % 2 == 1:
    mn = min(mn, a)
    sm += a
if mn == 100:
  print(-1)
else:
  print(sm)
  print(mn)
```
