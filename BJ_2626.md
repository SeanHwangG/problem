#include <iostream>
#include <cmath>
using namespace std;
int main() {
  int n, j; double c[2000], d[2000], x=0, y=0, k, t=1, it=5e5;
  cin >>n;
  for (int i=0; i<n; i++) cin >>c[i] >>d[i];
  while(it--) {
    k = 0;
    for (int i=0; i<n; i++) {
      double dist = (c[i]-x)*(c[i]-x) + (d[i]-y)*(d[i]-y);
      if (dist > k) k=dist, j=i;
    }
    x=(x+t*c[j])/(1+t);
    y=(y+t*d[j])/(1+t);
    t*=0.9993;
  }
  cout.precision(3); cout <<fixed <<x <<" " <<y <<endl <<sqrt(k);
}
