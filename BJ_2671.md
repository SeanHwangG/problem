```go
package main

import (
  "fmt"
  "regexp"
)

func main() {
  p, s := regexp.MustCompile("^(100+1+|01)+$"), ""
  fmt.Scan(&s)
  if p.MatchString(s) {
    fmt.Println("SUBMARINE")
  } else {
    fmt.Println("NOISE")
  }
}
```
