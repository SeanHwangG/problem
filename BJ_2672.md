```py
x = [[] for _ in range(10 ** 4)]
for _ in range(int(input())):
  l = [int(float(x) * 10) for n in input().split())]
  for i in range(l[0], l[0] + l[2]):
    x[i] += [(l[1], l[1] + l[3])]

s = 0
for i in x:
  if not i: continue
  i.sort()
  r=0
  for j in i:
    if j[1] <= r: continue
    s += j[1] - max(r,j[0])
    r = j[1]
print(s // 100 if s % 100 == 0 else s / 100)
```
