package main

import (
  "fmt"
)

func main() {
  var r, t int
  var s string

  fmt.Scan(&t)
  for i := 0; i < t; i++ {
    fmt.Scan(&r, &s)
    for j := 0; j < len(s); j++ {
      for k := 0; k < r; k++ {
        fmt.Printf("%c", s[j])
      }
    }
    fmt.Println()
  }
}
