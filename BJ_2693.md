#include <bits/stdc++.h>
using namespace std;

int main() {
  int T; cin >> T;

  array<int, 10> v;
  for (int t = 0; t < T; ++t) {
    for (int i = 0; i < 10; ++i)
      cin >> v[i];

    sort(v.rbegin(), v.rend());
    cout << v[2] << '\n';
  }
}
