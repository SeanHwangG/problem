```py
from bisect import insort
for _ in range(int(input())):
  m = int(input())
  print(str((m+1)//2))
  nums = []
  for t in range((m + 9) // 10):
    for i, n in enumerate(map(int,input().split())):
      insort(nums, n)
      if i % 2 == 0:
        print(nums[len(nums) // 2], end=' ')
    if t % 2 == 1:
      print()
  if t%2==0:
    print()
```
