```cpp
#include <iostream>
int main(){
  int i, n;
  for (std::cin>>n; i++ < 9; )
  printf("%d * %d = %d ", n, i, n * i);
}
```

```go
package main
import . "fmt"

func main() {
  var n int
  Scan(&n)
  for i := 1; i <= 9; i++ {
    Printf("%d * %d = %d\n", n, i, n*i)
  }
}
```

```js
var input = require('fs').readFileSync('/dev/stdin').toString().trim()
```
