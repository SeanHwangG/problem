```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
  int T; cin >> T;
  while (T--) {
    unordered_set<char> result;
    string x; cin >> x;
    for (int i = 0; i < x.size(); i++)
      result.insert(x[i]);
    cout << result.size() << '\n';
  }
  return 0;
}
```
