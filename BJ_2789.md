```go
package main

import (
  "bufio"
  "fmt"
  "os"
  "strings"
)

func main() {
  str := "CAMBRIDGE"
  var input string
  reader := bufio.NewReader(os.Stdin)
  fmt.Fscanln(reader, &input)
  for _, i := range str {
    input = strings.Replace(input, string(i), "", -1)
  }
  fmt.Println(input)
}
```

```py
a = "CAMBRIDGE"
s = input()
for i in a:

    if i == s[j]:
      s[j] = ''
```
