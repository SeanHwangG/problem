from itertools import combinations
def f(i):
  v, w = 1, 0
  for x, y in i:
    v *= x
    w += y
  return abs(v-w)
n,r=int(input()),range
l = [[*map(int,input().split())]for i in r(n)]
print(min(f(j) for i in r(n) for j in combinations(l, i + 1)))
