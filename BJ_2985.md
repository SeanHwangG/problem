* Given Three integers, find equation using operations

```txt
Input: 5 3 8
Output: 5+3=8
```

{% endtab %}{% tab title='BJ_2985.go' %}

```go
package main

import (
  "fmt"
)

func main() {
  var a, b, c int
  fmt.Scan(&a, &b, &c)

  switch {
  case a+b == c:
    fmt.Printf("%v+%v=%v", a, b, c)
  case a-b == c:
    fmt.Printf("%v-%v=%v", a, b, c)
  case a*b == c:
    fmt.Printf("%v*%v=%v", a, b, c)
  case a/b == c:
    fmt.Printf("%v/%v=%v", a, b, c)
  case a == b+c:
    fmt.Printf("%v=%v+%v", a, b, c)
  case a == b-c:
    fmt.Printf("%v=%v-%v", a, b, c)
  case a == b*c:
    fmt.Printf("%v=%v*%v", a, b, c)
  case a == b/c:
    fmt.Printf("%v=%v/%v", a, b, c)
  }
}
```
