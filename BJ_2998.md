package main

import (
  "fmt"
)

func main() {
  table := map[string]int{
    "000": 0, "001": 1, "010": 2, "011": 3, "100": 4, "101": 5, "110": 6, "111": 7,
  }

  var input = ""
  fmt.Scan(&input)

  var r = 3 - (len(input) % 3)
  if r == 3 {
    r = 0
  }

  var s = ""
  for i := 0; i < r; i++ {
    s += "0"
  }
  s = s + input
  rs := []rune(s)
  result := make([]int, 0)
  for i := 0; i < len(rs); i += 3 {
    v := string(rs[i : i+3])
    result = append(result, table[v])
  }

  for _, v := range result {
    fmt.Print(v)
  }
  fmt.Println()
}
