```java
import java.util.*;
public class Main{
  public static void main(String args[]) {
    int A[] = {1, 1, 2, 2, 2, 8};
    Scanner sc = new Scanner(System.in);
    for (int i = 0; i < 6; i++)
      System.out.println(A[i] - sc.nextInt());
  }
}
```

```go
package main

import "fmt"

func main() {
  var a, b, c, d, e, f int
  fmt.Scan(&a, &b, &c, &d, &e, &f)
  fmt.Println(1-a, 1-b, 2-c, 2-d, 2-e, 8-f)
}
```
