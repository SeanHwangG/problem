```go
package main
import "fmt"

func main() {
  var x1, y1, x2, y2, x3, y3 int
  fmt.Scan(&x1, &y1, &x2, &y2, &x3, &y3)
  fmt.Print(x1 ^ x2 ^ x3, y1 ^ y2 ^ y3)
}
```

```js
let X = 0, Y = 0;

require('fs').readFileSync('/dev/stdin').toString().split('\n').forEach((line) => {
  const [x, y] = line.split(' ').map(Number);
  X ^= x;
  Y ^= y;
})

console.log(X, Y);
```
