from copy import copy
for _ in range(int(input())):
  total = int(input())
  li = list(map(int, input().split()))
  for d in range(1, 100000):
    if sum(li) > total:
      print(d)
      break
    old = copy(li)
    for i in range(6):
      li[i] = old[i] + old[(i + 1) % 6] + old[(i + 3) % 6] + old[(i + 5) % 6]
