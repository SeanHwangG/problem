package main

import (
  "bufio"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func nextByte() byte {
  in.Scan()
  return in.Bytes()[0]
}

func main() {
  in.Split(bufio.ScanWords)
  for t := nextInt(); t > 0; t-- {
    main2()
  }
}

func main2() {
  r1, r2 := 0, 0
  for n := nextInt(); n > 0; n-- {
    a, b := nextByte(), nextByte()
    if a == 'R' && b == 'S' || a == 'P' && b == 'R' || a == 'S' && b == 'P' {
      r1++
    }
    if a == 'R' && b == 'P' || a == 'P' && b == 'S' || a == 'S' && b == 'R' {
      r2++
    }
  }
  switch {
  case r1 > r2:
    fmt.Println("Player 1")
  case r1 < r2:
    fmt.Println("Player 2")
  default:
    fmt.Println("TIE")
  }
}
