```py
while True:
  line = input()
  if line == '.':
    break
  stk = []
  for ch in line:
    if ch == '(' or ch == '[':
      stk.append(ch)
    elif ch == ')':
      if not stk or stk[-1] == '[':
        print("no")
        break
      elif stk[-1] == '(':
        stk.pop()
    elif ch == ']':
      if not stk or stk[-1] == '(':
        print("no")
        break
      elif stk[-1] == '[':
        stk.pop()
  else:
    print("no" if stk else "yes")
```
