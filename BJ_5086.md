```go
package main
import "fmt"

func main() {
  for {
    var a, b int
    fmt.Scan(&a, &b)
    if a == 0
      break
    if a % b == 0
      fmt.Println("multiple")
    else if b % a == 0
      fmt.Println("factor")
    else
      fmt.Println("neither")
  }
}
```

```py
while True:
  a, b = map(int, input().split())
  if a == 0 and b == 0:
    break
  if b % a == 0:
    print('factor')
  elif a % b == 0:
    print('multiple')
  else:
    print('neither')
```
