def operate(com, li):
  left = True
  if len(li) < com.count('D'):
    return 'error'
  for c in com:
    if c == 'R':
      left = not left
    elif c == 'D':
      li.pop(0 if left else -1)
  if left:
    return f"[{','.join(li)}]"
  else:
    return f"[{','.join(reversed(li))}]"

T = int(input())
for _ in range(T):
  com = input()
  n = input()
  li = input()[1:-1].split(',')
  if n=='0': li = []
  print(operate(com, li))
