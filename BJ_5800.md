```cpp
#include <bits/stdc++.h>

using namespace std;
int arr[50];

int main() {
  int t;
  scanf("%d", &t);

  for (int tt = 1; tt <= t;++tt) {
    int n;
    cin >> n;
    for (int i = 0; i < n;++i)
      cin >> arr[i];
    sort(arr, arr + n);

    int mx = 0;
    for (int i = 1; i < n; ++i)
      mx = max(mx, arr[i] - arr[i - 1]);

    printf("Class %d\nMax %d, Min %d, Largest gap %d\n", tt, arr[n - 1], arr[0], mx);
  }

  return 0;
}
```
