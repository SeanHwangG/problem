#include <bits/stdc++.h>
using namespace std;

int n, ans, sum, cas = 1, avg, h[50];

int main() {
  while (cin >> n && n) {
    sum = ans = 0;
    for (int i = 0; i < n; i++)
      cin >> h[i], sum += h[i];
    avg = sum / n;
    for (int i = 0; i < n; i++)
      if (h[i] > avg)
        ans += h[i] - avg;
    printf("Set #%d\nThe minimum number of moves is %d.\n\n", cas++, ans);
  }
  return 0;
}
