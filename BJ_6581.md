```py
import sys
lines = sys.stdin.read().split()
for i, line in enumerate(lines):
  if line == "<br>":
    print()
  elif line == "<hr>":
    if i != 0 and lines[i - 1] not in ("<br>", "<hr>"):
      print()
    print("-" * 80)
  else:
    print(line, end=" ")
```
