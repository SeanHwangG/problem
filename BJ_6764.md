using namespace std;
int main() {
  int a, b, c, d; cin >> a >> b >> c >> d;
  if (a == b && b == c && c == d) cout << "Fish At Constant Depth";
  else if (a < b && b < c && c < d) cout << "Fish Rising";

}

package main

import "fmt"

func main() {
  var a, b, c, d int
  fmt.Scan(&a, &b, &c, &d)
  r := "No Fish"
  switch {
  case a < b && b < c && c < d:
    r = "Fish Rising"
  case a > b && b > c && c > d:
    r = "Fish Diving"
  case a == b && b == c && c == d:
    r = "Fish At Constant Depth"
  }
  fmt.Print(r)
}
