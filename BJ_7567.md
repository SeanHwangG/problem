```go
package main

import "fmt"

func main() {
  var s string
  fmt.Scan(&s)
  p := ' '
  r := 0
  for _, c := range s {
    if p != c {
      r += 10
    } else {
      r += 5
    }
    p = c
  }
  fmt.Println(r)
}
```
