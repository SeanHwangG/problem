a, b = input().split()
def is_prime(n):
  if n < 2:
    return False
  for i in range(2, int(n ** 0.5) + 1):
    if n % i == 0:
      return False
  return True
print("Yes" if is_prime(int(a)) and is_prime(int(b + a)) else "No")
