def cost(arr, x):
  ret = 0
  for i in range(len(arr)):
    ret += abs(i * x - arr[i])
  return ret

N = int(input())
arr = list(map(int, input().split()))

lo, hi = 1, int(1e9)

while lo + 3 <= hi:
  l, r = (2 * lo + hi) // 3, (lo + 2 * hi) // 3

  mid1, mid2 = cost(arr, l), cost(arr, r)
  if mid1 < mid2:
    hi = r
  else:
    lo = l

ans = 1e18
for i in range(lo, hi+1):
  ans = min(ans, cost(arr, i))

print(ans)
