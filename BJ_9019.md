```py
import collections

for __ in range(int(input())):
  f, t = map(int, input().split())
  v = {f: (None, None)}
  q = collections.deque(v)
  while q and t not in v:
    x = q.popleft()
    for n, c in zip((x * 2 % 10000, (x - 1) % 10000, x * 10 % 10000 + x // 1000, x // 10 + x % 10 * 1000), 'DSLR'):
      if n not in v:
        v[n] = (x, c)
        q.append(n)
  r = ''
  x, c = v[t]
  while c:
    r += c
    x, c = v[x]
  print(r[::-1])
```
