#include <stdio.h>
#include <algorithm>
#include <queue>
using namespace std;
struct Mat {
  vector<vector<double> > data;
  Mat(int n) { data.resize(n, vector<double>(n, 0)); }
  vector<double> &operator[](int idx) {
    return data[idx];
  }
  void Swap(int i, int j) {
    if (i == j)
      return;
    swap(data[i], data[j]);
  }
};
int main() {
  int n;
  scanf("%d", &n);
  Mat a(n), b(n);
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++)
      scanf("%lf", &a[i][j]);
    b[i][i] = 1;
  }
  for (int k = 0; k < n; k++) {
    int t = k - 1;
    while (t + 1 < n && !a[++t][k]);
    if (t == n - 1 && !a[t][k])
      return !printf("no inverse");
    a.Swap(k, t), b.Swap(k, t);
    double d = a[k][k];

    for (int j = 0; j < n; j++)
      a[k][j] /= d, b[k][j] /= d;
    for (int i = 0; i < n; i++)
      if (i != k) {
        double m = a[i][k];
        for (int j = 0; j < n; j++) {
          if (j >= k)
            a[i][j] -= a[k][j] * m;
          b[i][j] -= b[k][j] * m;
        }
      }
  }
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < n; j++)
      printf("%.7f ", b[i][j]);
    puts("");
  }
}
