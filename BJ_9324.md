for _ in range(int(input())):
  s = input()
  c = [0] * 26
  for i in range(len(s)):
    x = s[i]
    j = ord(x)-65
    c[j] += 1
    if c[j] % 4 == 3:
      if i == len(s) - 1 or s[i + 1] != x:
        print('FAKE')
        break
  else:
    print('OK')
