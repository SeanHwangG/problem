```py
for i in range(int(input())):
  a, b, c = sorted(list(map(int, input().split())))

  if a + b <= c:
    ans = 'invalid!'
  elif a == b and b == c:
    ans = 'equilateral'
  elif a == b or b == c:
    ans = 'isosceles'
  else:
    ans = 'scalene'
  print(f'Case #{i + 1}: {ans}')
```
