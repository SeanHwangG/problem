```py
from bisect import *
import sys;input=sys.stdin.readline
nums=[]
N, K = map(int,input().split())
G = [int(input()) for i in range(N)]
ret = 0
kk = (K - 1) // 2
for i in range(N):
  insort(nums, G[i])
  if i < K - 1:
    continue
  if i >= K:
    nums.pop(bisect(nums, G[i - K])-1)
  ret += nums[kk]
print(ret)
```
