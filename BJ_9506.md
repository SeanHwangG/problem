def divisors(num):
  li = [1]
  for i in range(2, int(num ** 0.5 + 1)):
    if num % i == 0:
      li.append(i)
      if i != num // i:
        li.append(num // i)
  return sorted(li)
while True:
  n = int(input())
  if n == -1:
    break
  divs_n = divisors(n)
  if sum(divs_n) == n:
    print(n, '=', ' + '.join(map(str, divs_n)))
  else:
    print(f"{n} is NOT perfect.")
