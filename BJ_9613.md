from itertools import combinations
def GCD(a, b):
  while(b):
    a, b = b, a % b
  return a
n = int(input())
for _ in range(n):
  li = list(map(int, input().split()))
  print(sum(GCD(a, b) for a, b in combinations(li[1:], 2)))
