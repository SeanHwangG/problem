```py
from collections import defaultdict

for _ in range(int(input())):
  n,m=map(int,input().split())
  d=defaultdict(int)
  i_days= [(i, list(map(int,input().split()))[1:]) for i in range(m)]
  i_days.sort(key=lambda day : len(day[1]))
  ans= [0] * m
  for i, days in i_days:
    for day in days:
      if d[day] + 1 <= (m + 1) // 2:
        ans[i] = day
        d[day] += 1
        break
    else:
      print("NO")
      break
  else:
    print("YES")
    print(*ans)
```
