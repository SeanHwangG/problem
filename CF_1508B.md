```py
for _ in range(int(input())):
  n, k = map(int, input().split())
  b = bin(k - 1)[2:]

  if k - 1 > 0:
    b += '0'

  if len(b) > n:
    print(-1)
    continue

  b = b.rjust(n, '0')
  p, temp = [], []
  for i in range(n):
    temp += [str(i+1)]
    if b[i] == '0':
      p += temp[::-1]
      temp = []
  print(*p)
```
