```py
import itertools

n, m = map(int, input().split())
s = input()
periods = list(itertools.permutations('abc'))
c = {p: [0] for p in periods}
for i, ch in enumerate(s):
  for p in periods:
    c[p].append(c[p][-1] + (ch != p[i % 3]))
for _ in range(m):
  l, r = map(int, input().split())
  print(min(c[p][r] - c[p][l - 1] for p in periods))
```
