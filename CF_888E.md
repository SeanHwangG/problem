```py
from bisect import bisect_left
n, m = map(int, input().split())
t = list(map(int, input().split()))
def g(x, i, n, s):
  if i < n:
    g(x, i + 1, n, s + t[i])
    g(x, i + 1, n, s)
  else:
    x.add(s % m)
x, y = set(), set()
g(x, 0, n // 2, 0)
g(y, n // 2, n, 0)
y = sorted(y)
print(max(i + y[bisect_left(y, m - i) - 1] for i in x))
```
