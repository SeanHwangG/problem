# [BJ_1541](https://acmicpc.net/problem/1541)

```en
Given expression with +, -, and parentheses, make the value of this expression minimal by parentheses
```

```txt
Input: 55-50+40
Output: -35
```

## Solution

* go

  ```go
  package main

  import (
    "fmt"
    "strconv"
  )

  func main() {
    var T string
    var ans int = 0
    var minus int = 1
    var j string = ""

    fmt.Scanf("%s",&T)

    for i := range T {
      if T[i] == '+' {
        conv,_ := strconv.Atoi(j)
        ans += conv*minus
        j = ""
      } else if T[i] == '-' {
        conv,_ := strconv.Atoi(j)
        ans += conv*minus
        minus = -1
        j = ""
      } else {
        j += string(T[i])
      }
    }
    conv,_ := strconv.Atoi(j)
    ans += conv*minus
    fmt.Println(ans)
  }
  ```

* py

  ```py
  equations = input().split('-')
  ret = 0
  for i, equation in enumerate(equations):
    for num in equation.split('+'):
      ret += int(num) if i == 0 else -int(num)
  print(ret)
  ```
