for _ in range(int(input())):
  n, m, w = map(int, input().split())
  A = []
  for _ in range(m):
    s, e, t = map(int, input().split())
    A += [(s, e, t), (e, s, t)]
  for _ in range(w):
    s, e, t = map(int, input().split())
    A += [(s, e, -t)]

  dp = [0] * (n + 1)
  cycle = False
  for i in range(n):
    for s, e, t in A:
      if dp[e] > dp[s] + t:
        dp[e] = dp[s] + t
        if i == n - 1:
          cycle = True
  if cycle:
    print("YES")
  else:
    print("NO")
