from collections import deque

def bfs(n, m):
  dq, node2dist, node2head = deque([n]), {n : 0}, {}
  while len(dq) != 0:
    v = dq.popleft()
    for adj in [v + 1, v - 1, v * 2]:
      if adj not in node2dist and adj <= 100000:
        node2head[adj] = v
        node2dist[adj] = node2dist[v] + 1
        dq.append(adj)
    if v == m:
      return node2head

n, m = map(int, input().split())
path = [m]
cur = m
node2head = bfs(n, m)
while cur != n:
  cur = node2head[cur]
  path.append(cur)
print(len(path) - 1)
print(*reversed(path))
