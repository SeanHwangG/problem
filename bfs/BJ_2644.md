from collections import deque
def bfs(G, start, end):
  dq, visited = deque([(start, 0)]), set([start])
  while len(dq) != 0:
    v, dist = dq.popleft()
    if v == end:
      return dist
    for adj in G[v]:
      if adj not in visited:
        visited.add(adj)
        dq.append((adj, dist + 1))
  return -1
N = int(input())
G = [[] for _ in range(N)]
a, b = map(int, input().split())
for _ in range(int(input())):
  u, v = map(int, input().split())
  G[u - 1].append(v - 1)
  G[v - 1].append(u - 1)

print(bfs(G, a - 1, b - 1))
