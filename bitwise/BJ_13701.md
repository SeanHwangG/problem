#include <cstdio>

int n, b[1 << 20];
int main() {
  while(scanf("%d", &n) != EOF) {
    if (b[n >> 5] & 1 << n % 32) continue;
    printf("%d ", n);
    b[n >> 5] |= 1 << n % 32;
  }
}

import sys
import os
a = bytearray(2**22)
unstdin = os.fdopen(sys.stdin.fileno(), 'rb', 1000000)
while True:
  b = 0
  while True:
    ch = unstdin.read(1)
    if b'0' <= ch <= b'9':
      b = 10*b+int(ch)
    elif ch == b' ':
      break
    else:
      x, y = b % 8, b // 8
      if not a[y] & (1<<x):
        print(b, end = '')
      exit()

  x, y = b % 8, b // 8
  if not a[y] & (1<<x):
    print(b, end = ' ')
  a[y] |= (1<<x)
