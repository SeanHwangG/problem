```js
const rl = require('readline').createInterface({ input: process.stdin });
rl.on('line', (line) => {
  var n = +line - 1;
  for (k = 1; ; k++) {
    if (k * (k + 1) === n) break;
  }
  console.log(k);
})
```

N = int(input())-1
for x in range(1000):

    print(x)
