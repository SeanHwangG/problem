```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var min = 0
var inCars = 0
var res = 0

func main() {
  stdin := bufio.NewReader(os.Stdin)
  fmt.Fscanf(stdin, "%d\n", &min)
  fmt.Fscanf(stdin, "%d\n", &inCars)
  for i := 0; i < min; i++ {
    inCar := 0
    outCar := 0
    fmt.Fscanf(stdin, "%d %d\n", &inCar, &outCar)
    inCars += inCar
    inCars -= outCar
    if inCars < 0 {
      res = 0
      break
    }
    if res < inCars {
      res = inCars
    }
  }

  fmt.Println(res)
}
```
