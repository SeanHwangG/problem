```py
for _ in range(int(input())):
  n, m = map(int,input().split())
  G = [list(sorted(map(int,input().split()))) for i in range(n)]
  mins = [[10 ** 9, i] for i in range(m)]
  for i in range(n):
    o = list(sorted(mins, reverse=True))
    ans = [0] * m
    for j in range(m):
      mins[o[j][1]][0] = min(o[j][0], G[i][j])
      ans[o[j][1]] = G[i][j]
    print(*ans)
```
