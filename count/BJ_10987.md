```cpp
#include <cstdio>
int cnt;
char c;
int main() {
  while(scanf("%c", &c) != EOF){
    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') cnt++;
  }
  printf("%d", cnt);
  return 0;
}
```

```go
package main

import "fmt"

func main() {
  var s []byte
  fmt.Scan(&s)
  r := 0
  for _, c := range s {
    switch c {
    case 'a', 'e', 'i', 'o', 'u':
      r++
    }
  }
  fmt.Println(r)
}
```
