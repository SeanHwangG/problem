```go
package main

import (
  "bufio"
  "bytes"
  "fmt"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func next() []byte {
  in.Scan()
  return in.Bytes()
}

func main() {
  t := append([]byte(nil), next()...)
  r := 0
  for n := nextInt(); n > 0; n-- {
    v := next()
    if bytes.Contains(append(v[:len(v):len(v)], v...), t) {
      r++
    }
  }
  fmt.Println(r)
}
```
