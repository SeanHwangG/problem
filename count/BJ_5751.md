```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

func main() {
  in := bufio.NewReader(os.Stdin)
  out := bufio.NewWriter(os.Stdout)
  defer out.Flush()

  for {
    var n int
    fmt.Fscan(in, &n)
    if n == 0 {
      break
    }

    a, mary, john := 0, 0, 0
    for i := 0; i < n; i++ {
      fmt.Fscan(in, &a)

      if a == 0 {
        mary++
      } else {
        john++
      }
    }

    fmt.Fprintf(out, "Mary won %d times and John won %d times\n", mary, john)
  }
}
```
