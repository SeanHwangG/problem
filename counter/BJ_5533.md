```py
from collections import Counter
N = int(input())
G = []
c1, c2, c3 = Counter(), Counter(), Counter()
for _ in range(N):
  a, b, c = map(int, input().split())
  c1[a] += 1
  c2[b] += 1
  c3[c] += 1
  G.append([a, b, c])
for a, b, c in G:
  print((a if c1[a] == 1 else 0) + (b if c2[b] == 1 else 0) + (c if c3[c] == 1 else 0))
```
