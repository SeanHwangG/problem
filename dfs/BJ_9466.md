```py
for _ in range(int(input())):
  n = int(input())
  choice = list(n - 1 for n in map(int, input().split()))
  visit = [0] * n
  group = 1
  for i in range(n):
    while visit[i] == 0:
      visit[i] = group
      i = choice[i]
    while visit[i] == group:
      visit[i] = -1
      i = choice[i]
    group += 1
  print(n - visit.count(-1))
```
