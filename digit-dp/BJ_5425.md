def sum1ToN(n):
  if n < 10:
    return n * (n + 1) // 2
  a, b = int(str(n)[0]), int(str(n)[1:])
  N = len(str(n)) - 1
  return (45 * a * N + a * (a - 1) // 2 * 10) * (10 ** (N - 1)) + a * (b + 1) + sum1ToN(b)
n = int(input())

for _ in range(n):
  a, b = map(int, input().split())
  print(sum1ToN(b) - sum1ToN(a - 1))
