#include <cstdio>

using namespace std;

int parent[1000001];
int n;

int find(int x) {
  if (x == parent[x]) return x;
  return parent[x] = find(parent[x]);
}

void union_(int x, int y) {
  int px = find(x);
  int py = find(y);

  if (px > py) parent[px] = py;
  else if (px < py) parent[py] = px;
}

int main() {
  int m;

  scanf("%d %d", &n, &m);

  for (int i = 0; i <= n; i++) parent[i] = i;

  for (int i = 0; i < m; i++) {
    int op, x, y;
    scanf("%d %d %d", &op, &x, &y);

    if (op == 0) {
      union_(x, y);
    } else {
      if (find(x) == find(y)) printf("YES\n");
      else printf("NO\n");
    }
  }
  return 0;
