Store which upper line is filled in binary

```py
n, m = map(int, input().split())
d=[[0] * (1 << m) for i in range(n * m + 1)]
d[n * m][0] = 1
for i in range(n * m - 1,- 1, -1):
  for j in range(1 << m):
    if j & 1:
      d[i][j] = d[i + 1][j >> 1]
      continue
    if i < (n - 1) * m:
      d[i][j] += d[i + 1][(1 << (m - 1)) | (j >> 1)]
    if i % m < m - 1 and not j % 4:
      d[i][j] += d[i + 2][j >> 2]
print(d[0][0] % 9901)
```
