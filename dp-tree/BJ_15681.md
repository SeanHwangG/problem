```py
import sys
sys.setrecursionlimit(10 ** 9)
input = sys.sdin.readline

N, R, Q = list(map(int,input().split()))
G = [[] for i in range(N + 1)]
dp = [0] * (N + 1)
for i in range(N-1):
  a, b = map(int, input().split())
  G[a].append(b)
  G[b].append(a)

def dfs(n):
  dp[n] = 1
  for node in G[n]:
    if not dp[node]:
      dfs(node)
      dp[n] += dp[node]

dfs(R)
for i in range(Q):
  print(dp[int(input())])
```
