input = sys.stdin.readline
sys.setrecursionlimit(1000000000)

N = int(input())
V = [0] + list(map(int,input().split()))
G = [[] for i in range(N + 1)]
for i in range(N - 1):
  a, b = map(int, input().split())
  G[a].append(b)
  G[b].append(a)
dp = [[0, 0] for i in range(N + 1)] # dp[v][0]: without v, dp[v][1]: with v

def dfs(u, p):
  dp[u][1] = V[u]
  for v in G[u]:
    if v == p: continue
    dfs(v, u)
    dp[u][0] += max(dp[v][0], dp[v][1])
    dp[u][1] += dp[v][0]

def track(u, p, last):
  if not last:
    if dp[u][0] < dp[u][1]:
      l.append(u)
      last = True
  else:
    last = False
  for v in G[u]:
    if v != p:
      track(v, u, last)
