# [BJ_3028](https://acmicpc.net/problem/3028)

* en

  ```en
  Print location of cup at the end of simulation
  A reverse 1, 2 / B reverse 2, 3 / C reverse 1, 3
  ```

* tc

  ```tc
  Input: AB
  Output: 3

  Input: CBABCACCC
  Output: 1
  ```

## Solution

* cpp

  ```cpp
  #include <bits/stdc++.h>

  using namespace std;

  int main() {
    int a, b, c;
    a = 1, b = c = 0;
    string s;
    cin >> s;

    for (auto i : s) {
      if (i == 'A') swap(a, b);
      if (i == 'B') swap(b, c);
      if (i == 'C') swap(a, c);
    }
    if (a) cout << 1 << '\n';
    if (b) cout << 2 << '\n';
    if (c) cout << 3 << '\n';
  }
  ```

* go

  ```go
  package main

  import (
    "bufio"
    "fmt"
    "os"
  )

  var reader *bufio.Reader = bufio.NewReader(os.Stdin)
  var writer *bufio.Writer = bufio.NewWriter(os.Stdout)

  func printf(f string, a ...interface{}) { fmt.Fprintf(writer, f, a...) }
  func scanf(f string, a ...interface{})  { fmt.Fscanf(reader, f, a...) }

  func main() {
    defer writer.Flush()

    var s string
    scanf("%s", &s)

    q := [3]int{1, 0, 0}
    for _, c := range s {
      switch c {
      case 'A':
        q[0], q[1] = q[1], q[0]
      case 'B':
        q[1], q[2] = q[2], q[1]
      case 'C':
        q[0], q[2] = q[2], q[0]
      }
    }

    for i, qq := range q {
      if qq == 1 {
        printf("%d", i+1)
      }
    }
  }
  ```
