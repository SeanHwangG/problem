is_prime = [False, False, True] + [True] * (123456 * 2)
for i in range(2, len(is_prime)):
  if is_prime[i]:
    for j in range(2 * i, len(is_prime), i):
      is_prime[j] = False

while True:
  N = int(input())
  if N == 0:
    break

  count = 0
  for n in range(N + 1, N * 2 + 1):
    if is_prime[n]:
      count += 1
  print(count)
