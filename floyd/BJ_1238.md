```py
N, M, X = map(int, input().split())
X -= 1
MAX = 10000000
dist = [[MAX for _ in range(N)] for _ in range(N)]
for _ in range(M):
  u, v, d = map(int, input().split())
  dist[u - 1][v - 1] = d

for i in range(N):
  dist[i][i] = 0

for j in range(N):
  for i in range(N):
    if dist[j][i] == MAX: continue
    for k in range(N):
      dist[k][i] = min(dist[k][i], dist[k][j] + dist[j][i])

print(max(dist[i][X] + dist[X][i] for i in range(N)))
```
