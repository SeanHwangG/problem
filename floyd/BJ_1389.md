```py
V, E = map(int, input().split())
G = [[float('inf')] * V for _ in range(V)]
for i in range(V):
  G[i][i] = 0
for _ in range(E):
  u, v = map(int, input().split())
  G[u - 1][v - 1] = 1
  G[v - 1][u - 1] = 1

for i in range(V):
  for j in range(V):
    for k in range(V):
      if G[j][i] + G[i][k] < G[j][k]:
        G[j][k] = G[j][i] + G[i][k]
bacon_scores = [sum(li) for li in G]
print(bacon_scores.index(min(bacon_scores)) + 1)
```
