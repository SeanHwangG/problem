```js
var input = require('fs').readFileSync('/dev/stdin').toString().trim().split('\n').map((e) => parseInt(e));
var n = input.shift();
console.log('Junhee is ' + (input.reduce((prev, curr) => prev + curr) > (n / 2) ? '' : 'not ') + 'cute!');
```
