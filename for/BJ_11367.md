```py
for i in range(int(input())):
  name, score = input().split()
  print(name, end=" ")
  score = int(score)
  if score >= 97:
    print("A+")
  elif score >= 90:
    print("A")
  elif score >= 87:
    print("B+")
  elif score >= 80:
    print("B")
  elif score >= 77:
    print("C+")
  elif score >= 70:
    print("C")
  elif score >= 67:
    print("D+")
  elif score >= 60:
    print("D")
  else:
    print("F")
```
