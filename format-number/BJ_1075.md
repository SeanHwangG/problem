```java
import java.util.*;
public class Main {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int n = scanner.nextInt(), d = scanner.nextInt();
    scanner.close();
    n -= n % 100;
    while (n++ % d != 0);
    n--;
    System.out.printf("%02d", n % 100);
  }
}
```
