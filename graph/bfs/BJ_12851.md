# [BJ_12851](https://acmicpc.net/problem/12851)

Given two positions X and Y, and after 1 second, X can move to positions X-1, X+1, and 2*X
How many seconds does it take to go Y, and how many shortest ways are there

```txt
Input: 5 17
Output: 4
2
```

## Solution

* go

  ```go
  package main

  import "fmt"

  func main() {
    var n, k int
    var x [3]int
    fmt.Scan(&n, &k)
    if n == k {
      fmt.Println("0\n1")
      return
    }
    r := map[int]int{n: 1}
    s := map[int]int{n: 1}
    q := []int{n}
    for {
      c := q[0]
      q = q[1:]
      if c == k {
        fmt.Printf("%d\n%d\n", r[c]-1, s[c])
        break
      }
      x[0], x[1], x[2] = c-1, c+1, c*2
      y := x[:]
      if c > k {
        y = x[:1]
      } else if c == 0 {
        y = x[1:]
      }
      for _, d := range y {
        if r[d] == 0 {
          r[d] = r[c] + 1
          q = append(q, d)
        }
        if r[d] == r[c]+1 {
          s[d] += s[c]
        }
      }
    }
  }
  ```

* py

  ```py
  from collections import defaultdict

  n, k = map(int, input().split())
  T = 100001
  q, v, c = {n: 1}, [1e9] * T, 0
  while q:
    dq = defaultdict(int)
    for n in dq:
      if n == k:
        print(c, q[k])
        exit()
      for i in [n - 1, n + 1, n * 2]:
        if 0 <= i < T and c <= v[i]:
          v[i] = c
          dq[i] += q[n]
    q = dq
    c += 1
  ```
