# [BJ_1260](https://acmicpc.net/problem/1260)

```en
Output results of exploring graph with DFS and the results of exploring with BFS
```

```txt
Input: 4 5 1
1 2
1 3
1 4
2 4
3 4

Output: 1 2 4 3
1 2 3 4
```

## Solution

* java

  ```java
  import java.util.*;
  class Main{
    static int[][] map;
    static boolean[] vs;
    static void dfs(int v) {
      vs[v] = true;
      System.out.print(v + " ");
      for (int i = 1; i < map.length; i++) {
        if (!vs[i] && map[v][i] == 1) dfs(i);
      }
    }
    static void bfs(int v) {
      Queue<Integer> q = new LinkedList<>();
      q.add(v);
      vs[v] = true;
      while (!q.isEmpty()) {
        int a = q.poll();
        System.out.print((a == v? "\n": "") + a + " ");
        for (int i = 1; i < map.length; i++) {
          if (!vs[i] && map[a][i] == 1) {
            vs[i] = true;
            q.add(i);
          }
        }
      }
    }
    public static void main(String[] ar) {
      Scanner sc = new Scanner(System.in);
      int n = sc.nextInt(), m = sc.nextInt(), v = sc.nextInt();
      map = new int[n + 1][n + 1];
      for (int i = 0; i < m; i++) {
        int a = sc.nextInt(), b = sc.nextInt();
        map[a][b] = map[b][a] = 1;
      }
      vs = new boolean[n+1]; dfs(v);
      vs = new boolean[n+1]; bfs(v);
    }
  }
  ```

* py

  ```py
  from collections import deque
  def dfs(G, v, visited=None):
    visited = visited or set([v])
    print(v, end=' ')
    for neighbor in sorted(G[v]):
      if neighbor not in visited:
        visited.add(neighbor)
        dfs(G, neighbor, visited)
    return visited
  def bfs(G, start):
    dq, visited = deque([start]), set([start])
    while len(dq) != 0:
      v = dq.popleft()
      print(v, end=' ')
      for adj in sorted(G[v]):
        if adj not in visited:
          visited.add(adj)
          dq.append(adj)
  N, M, v = map(int, input().split())
  G = [[] for i in range(N+1)]
  for i in range(M):
    a, b = map(int, input().split())
    G[a].append(b)
    G[b].append(a)

  dfs(G, v)
  print()
  bfs(G, v)
  ```
