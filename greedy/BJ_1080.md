```py
N, M =map(int,input().split())
A = [list(map(int, list(input()))) for _ in range(N)]
B = [list(map(int, list(input()))) for _ in range(N)]
def flip(G, x,y):
  for i in range(x, x + 3):
    for j in range(y, y + 3):
      G[i][j] = 1 - G[i][j]
cnt = 0
for i in range(0, N - 2):
  for j in range(0, M - 2):
    if A[i][j] != B[i][j]:
      flip(A, i, j)
      cnt+=1
for i in range(N):
  for j in range(M):
    if A[i][j] != B[i][j]:
      print(-1)
      exit()
print(cnt)
```
