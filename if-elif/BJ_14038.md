```go
package main
import "fmt"

func main() {
  var s string
  var cnt int = 0
  for i := 0; i < 6; i++ {
    fmt.Scanf("%s", &s)
    if s == "W" {
      cnt += 1
    }
  }
  var res int
  if cnt >= 5 {
    res = 1
  } else if cnt >= 3 {
    res = 2
  } else if cnt >= 1 {
    res = 3
  } else {
    res = -1
  }
  fmt.Printf("%d\n", res)
}
```
