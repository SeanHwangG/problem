```go
package main

import "fmt"

func main() {
  var a, b int
  fmt.Scan(&a, &b)
  c := b - a
  switch {
  case c <= 0:
    fmt.Print("Congratulations, you are within the speed limit!")
  case c <= 20:
    fmt.Print("You are speeding and your fine is $100.")
  case c <= 30:
    fmt.Print("You are speeding and your fine is $270.")
  default:
    fmt.Print("You are speeding and your fine is $500.")
  }
}
```
