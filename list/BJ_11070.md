```py
for _ in range(int(input())):
  n, k = map(int, input().split())
  w, l = [0] * n, [0] * n
  for _ in range(k):
    a, b, c, d = map(int,input().split())
    w[a - 1] += c
    l[a - 1] += d
    w[b - 1] += d
    l[b - 1] += c
  c = [0] * n
  for i in range(n):
    t= w[i] * w[i]
    if w[i] + l[i] != 0:
      c[i] = t/(t+l[i]*l[i])*1000
  print(int(max(c)))
  print(int(min(c)))
```
