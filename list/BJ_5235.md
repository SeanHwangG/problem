```go
package main

import (
  "bufio"
  "os"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func main() {
  in.Split(bufio.ScanWords)
  for t := nextInt(); t > 0; t-- {
    os.Stdout.WriteString(main2())
  }
}

func main2() string {
  r1, r2 := 0, 0
  for n := nextInt(); n > 0; n-- {
    if v := nextInt(); v%2 == 0 {
      r2 += v
    } else {
      r1 += v
    }
  }
  switch {
  case r1 > r2:
    return "ODD\n"
  case r1 < r2:
    return "EVEN\n"
  }
  return "TIE\n"
}
```
