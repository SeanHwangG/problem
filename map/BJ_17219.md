```go
package main

import (
  "bufio"
  "fmt"
  "os"
)

var (
  reader *bufio.Reader
  writer *bufio.Writer
)

func init() {
  reader = bufio.NewReader(os.Stdin)
  writer = bufio.NewWriter(os.Stdout)
}

func main() {
  defer writer.Flush()

  data := map[string]string{}
  var n, m int
  var s, p string

  fmt.Fscan(reader, &n, &m)

  for i := 0; i < n; i++ {
    fmt.Fscan(reader, &s, &p)
    data[s] = p
  }

  for i := 0; i < m; i++ {
    fmt.Fscan(reader, &s)
    writer.WriteString(data[s] + "\n")
  }
}
```
