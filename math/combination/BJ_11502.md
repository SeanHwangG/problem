# [BJ_11502](https://acmicpc.net/problem/11502)

```en
Take any odd number greater than 5
Determine how that odd number can be expressed as a sum of three decimal places
```

```txt
Input:
3
7
11
25

Output:
2 2 3
2 2 7
5 7 13
```

## Solution

* go

  ```go
  package main

  import (
    "bufio"
    "fmt"
    "os"
    "sort"
  )

  var in = bufio.NewScanner(os.Stdin)
  var out = bufio.NewWriter(os.Stdout)
  var sieve [502]bool
  var primes [167]int

  func nextInt() int {
    in.Scan()
    r := 0
    for _, c := range in.Bytes() {
      r *= 10
      r += int(c - '0')
    }
    return r
  }

  func main() {
    var p int
    for i := 3; i < 1004; i += 2 {
      if !sieve[i/2] {
        primes[p] = i
        p++
        for j := i * i; j < 1004; j += i + i {
          sieve[j/2] = true
        }
      }
    }
    for t := nextInt(); t > 0; t-- {
      main2()
    }
    out.Flush()
  }

  func main2() {
    n := nextInt()
    if n == 7 {
      fmt.Fprintln(out, 2, 2, 3)
      return
    }
    n -= 3
    a, b := 0, sort.SearchInts(primes[:], n-primes[0])
    for {
      d := primes[a] + primes[b]
      switch {
      case n > d:
        a++
      case n < d:
        b--
      case n == d:
        fmt.Fprintln(out, 3, primes[a], primes[b])
        return
      }
    }
  }
  ```

* py

  ```py
  from itertools import combinations_with_replacement
  is_prime = [False, False] + [True] * (1000)
  primes = []
  for i in range(2, len(is_prime)):
    if is_prime[i]:
      primes.append(i)
      for j in range(2 * i, len(is_prime), i):
        is_prime[j] = False
  for _ in range(int(input())):
    n = int(input())
    for a, b, c in combinations_with_replacement(primes, 3):
      if a + b + c == n:
        print(a, b, c)
        break
    else:
      print(0)
  ```
