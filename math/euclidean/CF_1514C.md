## Solution

* cpp

  ```cpp
  #include <bits/stdc++.h>
  using namespace std;
  bool ok[100005];
  int main() {
    int n;
    cin >> n;
    long long prod=1;
    for (int i = 1; i < n; i++) {
      if (__gcd(n, i) == 1) {
        ok[i] = 1;
        prod = (prod  * i) % n;
      }
    }
    if (prod != 1) ok[prod] = 0;
    cout << count(ok + 1, ok + n, 1) << endl;
    for (int i = 1; i < n; i++) {
      if (ok[i]) cout << i;
    }
  }
  ```
