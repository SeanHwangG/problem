```py
import sys
input = sys.stdin.readline
h, m, s = map(int, input().split())
s += (h * 3600 + m * 60)

for i in range(int(input())):
  l = list(map(int, input().split()))
  if len(l) == 2:
    q, c = l
  else:
    q, c = l, 0

  if q == 1:
    s += c
  elif q == 2:
    s -= c
  else:
    print(s // 3600 % 24, (s % 3600) // 60, s % 60)
```
