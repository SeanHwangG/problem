```go
package main

import (
  "bufio"
  "fmt"
  "os"
  "strconv"
)

var in = bufio.NewScanner(os.Stdin)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func next() string {
  in.Scan()
  return in.Text()
}

func main() {
  in.Split(bufio.ScanWords)
  for t := nextInt(); t > 0; t-- {
    k, v := nextInt(), next()
    r1, _ := strconv.ParseInt(v, 8, 64)
    r2, _ := strconv.ParseInt(v, 10, 64)
    r3, _ := strconv.ParseInt(v, 16, 64)
    fmt.Println(k, r1, r2, r3)
  }
}
```
