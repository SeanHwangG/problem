def solve(n, arr):
  d = [arr[i] - arr[i-1] for i in range(1, n)]
  s = set(d)
  if len(s) <= 1:
    return 0
  if len(s) >= 3:
    return -1
  x, y = max(s), min(s)
  if x <= 0 or y >= 0 or x - y <= max(arr):
    return -1
  return f"{x - y} {x}"

import sys
input = lambda: sys.stdin.readline().rstrip()
t = int(input())
for i in range(t):
  n = int(input())
  arr = list(map(int, input().split()))
  print(solve(n, arr))
