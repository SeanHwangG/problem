_, length = map(int, input().split())
li = list(map(int, input().split()))
def total(li, cut):
  ret = 0
  for n in li:
    ret += max(0, n - cut)
  return ret
def binary_search(li, x):
  lo, hi = 0, max(li)
  while lo < hi:
    mi = (hi + lo + 1) // 2
    if total(li, mi) < x:
      hi = mi - 1
    else:
      lo = mi
  return lo
print(binary_search(li, length))
