# [BJ_2658](https://acmicpc.net/problem/2658)

Check if Grid is isosceles right triangle, if so print corner

```txt
Input: 0000000000
0000001000
0000011000
0000111000
0001111000
0000111000
0000011000
0000001000
0000000000
0000000000
Output:
2 7
5 4
8 7
```

## Solution

```cpp
#include <bits/stdc++.h>
#define MAX 15

int G[MAX][MAX];

pair<int, int> xy[3];
int n_corner, t;

bool is_filled() {
  int zero, flag;
  for(int i=1;i<=10;i++) {
    flag = zero = 0;
    for(int j=1;j<=10;j++) {
      if (G[i][j]) {
        if (flag) continue;
        else {
          if(zero) return false;
          flag = 1;
        }
      }
      else {
        if(flag) {
          flag=0;
          zero++;
        }
        else continue;
      }
    }
  }
  return true;
}

int neighbor_zero(int x, int y) {
    int cnt=0;
    for(int i=-1;i<=1;i++)
      for(int j=-1;j<=1;j++)
        if(!G[x+i][y+j]) cnt++;

    return cnt;
}

int main() {
  for(int i=1;i<=10;i++)
    for(int j=1;j<=10;j++)
      scanf("%1d", &G[i][j]);
  for(int i=1;i<=10;i++)
    for(int j=1;j<=10;j++)
      if(G[i][j]) {
        t = neighbor_zero(i, j);
        if (5 <= t and t < 8) xy[n_corner++] = {i, j};
      }

  if(!is_filled() or n_corner != 3) printf("0");
  else {
    sort(xy, xy + n_corner);
    for(int i=0;i<n_corner;i++) printf("%d %d\n", xy[i].first, xy[i].second);
  }
}
```
