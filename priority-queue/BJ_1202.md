import sys
from heapq import *
input = sys.stdin.readline

n, k = map(int, input().split())
info = sorted(list(map(int, input().split())) for _ in range(n))
bag = sorted(int(input()) for _ in range(k))
j = ans = 0
q = []
for b in bag:
  while j < n and info[j][0] <= b:
    heappush(q, -info[j][1])
    j += 1
  if q:
    ans -= heappop(q)
print(ans)
