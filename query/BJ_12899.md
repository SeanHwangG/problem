import sys
input = sys.stdin.readline

def update(pos, val):
  pos += N
  while pos > 0:
    seg[pos] += val
    pos >>= 1
  return

def search(k):
  i = 1
  while i < N:
    if seg[2*i] >= k:
      i *= 2
    else:
      k -= seg[2*i]
      i = 2*i + 1
  return i - N

N = 1 << 21
seg = [0] * 2*N
for _ in range(int(input())):
  t, x = map(int, input().split())
  if t == 1:
    update(x - 1, 1)
  if t == 2:
    tmp = search(x)
    update(tmp, -1)
    print(tmp + 1)
