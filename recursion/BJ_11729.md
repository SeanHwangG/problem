def hanoi(disk, start, mid, end):
  if disk == 1:
    moves.append([start, end])
  else:
    hanoi(disk - 1, start, end, mid)
    moves.append([start, end])
    hanoi(disk - 1, mid, start, end)
total_disk = int(input())
moves = []
hanoi(total_disk, 1, 2, 3)
print(len(moves))
for move in moves:
  print(move[0], move[1])
