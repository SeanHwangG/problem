# [BJ_16360](https://acmicpc.net/problem/16360)

Print string in following conversion table

| English | pseudo-Latin |
| ------- | ------------ |
| -a      | -as          |
| -i, -y  | -ios         |
| -l      | -les         |
| -n, -ne | -anes        |
| -o      | -os          |
| -r      | -res         |
| -t      | -tas         |
| -u      | -us          |
| -v      | -ves         |
| -w      | -was         |

```txt
Input:
2
toy
engine

Output:
toios
engianes
```

## Solution

```go
package main

import (
  "bufio"
  "bytes"
  "os"
)

var data = [][2][]byte{
  {[]byte("a"), []byte("as")},
  {[]byte("i"), []byte("ios")},
  {[]byte("y"), []byte("ios")},
  {[]byte("l"), []byte("les")},
  {[]byte("n"), []byte("anes")},
  {[]byte("ne"), []byte("anes")},
  {[]byte("o"), []byte("os")},
  {[]byte("r"), []byte("res")},
  {[]byte("t"), []byte("tas")},
  {[]byte("u"), []byte("us")},
  {[]byte("v"), []byte("ves")},
  {[]byte("w"), []byte("was")},
  {[]byte(""), []byte("us")},
}

var in = bufio.NewScanner(os.Stdin)
var out = bufio.NewWriter(os.Stdout)

func nextInt() int {
  in.Scan()
  r := 0
  for _, c := range in.Bytes() {
    r *= 10
    r += int(c - '0')
  }
  return r
}

func next() []byte {
  in.Scan()
  return in.Bytes()
}

func main() {
  for t := nextInt(); t > 0; t-- {
    main2()
  }
  out.Flush()
}

func main2() {
  s := next()
  for _, d := range data {
    if bytes.HasSuffix(s, d[0]) {
      out.Write(bytes.TrimSuffix(s, d[0]))
      out.Write(d[1])
      out.WriteByte('\n')
      return
    }
  }
}
```

```py
import re
for _ in range(int(input())):
  i = input()
  for v, w in[*zip(['[iy]','ne?','([aou])','([tw])','([rvl])',''],['ios','anes','\g<1>s','\g<1>as','\g<1>es','us'])]:
    if re.search(v + '$',i):
      i = re.sub(v + '$', w, i)
      break
  print(i)
```
