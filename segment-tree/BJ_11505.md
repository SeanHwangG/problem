import sys
input = sys.stdin.readline

def update(i, v):
  tree[i]=v
  while i>1:
    tree[i // 2] = (tree[i] * tree[i ^ 1]) % MOD
    i //= 2

def query(l,r):
  res = 1
  while l < r:
    if l & 1:
      res *= tree[l]
      l += 1
    if r & 1:
      r -= 1
      res *= tree[r]
    l >>= 1
    r >>= 1
  return res % MOD

MOD = 10 ** 9 + 7
n, m, k = map(int, input().split())
tree = [1] * (2 * n + 10)
for i in range(n + 1, 2 * n + 1):
  tree[i] = int(input())
for i in range(n, 0, -1):
  tree[i] = (tree[2 * i] * tree[2 * i + 1]) % MOD

for _ in range(m + k):
  a, b, c=map(int, input().split())
  if a==1:
    update(n+b,c)
  else:
    print(query(n+b,n+c+1))
