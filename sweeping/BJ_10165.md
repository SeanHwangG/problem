```py
import sys

N, M = int(input()), int(input())
li = []
for i in range(1, M + 1):
  l, h = map(int, sys.stdin.readline().split())
  if l < h:
    li.append([l, h, i])
  else:
    li.append([l - N, h, i])
    li.append([l, h + N, i])

li.sort(key = lambda x : (x[0], -x[1]))
print(*li, sep='\n')
res = set()
right = -1

for i in range(len(li)):
  if right < li[i][1]:
    right = li[i][1]
    res.add(li[i][2])
print(*sorted(res))
```
