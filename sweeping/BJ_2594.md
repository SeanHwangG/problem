```py
times = [(0, 600), (1320, 1440)]
for _ in range(int(input())):
  a, b = input().split()
  times.append((int(a[:2]) * 60 + int(a[2:]) - 10, int(b[:2]) * 60 + int(b[2:]) + 10))
times.sort()
mx, last_ended = 0, 600
for start, end in times:
  mx = max(mx, start - last_ended)
  last_ended = max(last_ended, end)

print(mx)
```
