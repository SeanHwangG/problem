# [BJ_5555](https://acmicpc.net/problem/5555)

```en
Print # rings containing the string you want to find
string is circular (as they are written on a ring) 
```

```txt
Input: ABCD
3
ABCDXXXXXX
CDYYAB
DCBAZZZZZZ

Output: 2  # There are ABCD in ABCDXXXXXX, CDYYAB
```

## Solution

* cpp

  ```cpp
  #include <iostream>
  using namespace std;
  int main() {
    string a;
    int n;
    cin >> a >> n;
    string b;
    int re = 0;
    for (int i = 0; i < n; i++) {
      cin >> b;
      string s = b + b;
      if (s.find(a) != string::npos) re++;
    }
    cout << re;
  }
  ```

* go

  ```go
  package main

  import (
    "bufio"
    "bytes"
    "fmt"
    "os"
  )

  var in = bufio.NewScanner(os.Stdin)

  func nextInt() int {
    in.Scan()
    r := 0
    for _, c := range in.Bytes() {
      r *= 10
      r += int(c - '0')
    }
    return r
  }

  func next() []byte {
    in.Scan()
    return in.Bytes()
  }

  func main() {
    t := append([]byte(nil), next()...)
    r := 0
    for n := nextInt(); n > 0; n-- {
      v := next()
      if bytes.Contains(append(v[:len(v):len(v)], v...), t) {
        r++
      }
    }
    fmt.Println(r)
  }
  ```
