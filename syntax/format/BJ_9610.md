# [BJ_9610](https://acmicpc.net/problem/9610)

* en

  ```en
  Count number of points on each coordinate and axis
  ```

* tc

  ```tc
  Input: 5
  0 0  # Axis
  0 1  # Axis
  1 1  # Quadrant 1
  3 -3 # Quadrant 4
  2 2  # Quddrant 1

  Output:
  Q1: 2
  Q2: 0
  Q3: 0
  Q4: 1
  AXIS: 2
  ```

## Solution

* go

  ```go
  package main

  import "fmt"

  func main() {
    var t int
    var q1, q2, q3, q4, axis int
    fmt.Scan(&t)

    for i := 0; i < t; i++ {
      var x, y int
      fmt.Scan(&x)
      fmt.Scan(&y)
      if x == 0 || y == 0 {
        axis++
      }
      if x > 0 && y > 0 {
        q1++
      }
      if x < 0 && y > 0 {
        q2++
      }
      if x < 0 && y < 0 {
        q3++
      }
      if x > 0 && y < 0 {
        q4++
      }
    }

    fmt.Printf("Q1: %d\n", q1)
    fmt.Printf("Q2: %d\n", q2)
    fmt.Printf("Q3: %d\n", q3)
    fmt.Printf("Q4: %d\n", q4)
    fmt.Printf("AXIS: %d", axis)
  }
  ```

* py

  ```py
  q1 = q2 = q3 = q4 = axis = 0
  N = int(input())
  for _ in range(N):
    x, y = map(int, input().split())
    if x == 0 or y == 0:
      axis += 1
    elif x > 0 and y > 0:
      q1 += 1
    elif x < 0 and y > 0:
      q2 += 1
    elif x < 0 and y < 0:
      q3 += 1
    elif x > 0 and y < 0:
      q4 += 1
  print(f"Q1: {q1}", f"Q2: {q2}", f"Q3: {q3}" f"Q4: {q4}" f"AXIS: {axis}", sep="\n")
  ```
