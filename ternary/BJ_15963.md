```asm
section  .bss
  a:     resq 1
  b:     resq 1
section  .data
  input: db "%d %d", 0
  yes:   db "1", 10, 0
  no:    db "0", 10, 0
section .text
  global  main
  extern  printf
  extern  scanf
main:
  push  rbp

  lea   rdx, [a]
  lea   rsi, [b]
  mov   rdi, input
  xor   rax, rax
  call  scanf

  mov   r8, [a]
  mov   r9, [b]
  cmp   r8, r9
  je    yeah
  jmp   nah
yeah:
  mov   rdi, yes
  xor   rax, rax
  call  printf
  ret
nah:
  mov   rdi, no
  xor   rax, rax
  call  printf
  ret
```
