from collections import deque
input = sys.stdin.readline
def bfs(G, x):

  dist[x] = 0
  while dq:
    x = dq.popleft()
    for w, nx in G[x]:
      if dist[nx] == -1:
        dist[nx] = dist[x] + w
        dq.append(nx)
  return max(dist), dist.index(max(dist))
n = int(input())
G = [[] for _ in range(n)]

  G[x-1].append([w, y-1])
  G[y-1].append([w, x-1])
