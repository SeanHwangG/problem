```py
costs = list(map(int, input().split()))
li = []
for _ in range(3):
  e, l = map(int, input().split())
  li.extend([(e, 'e'), (l, 'l')])

ret, cur, last = 0, 0, 0
for time, typ in sorted(li):
  if typ == 'e':
    ret += (time - last) * costs[cur - 1] * cur
    cur += 1
  else:
    ret += (time - last) * costs[cur - 1] * cur
    cur -= 1
  last = time

print(ret)
```
