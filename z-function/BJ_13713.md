def Z(s):
  z = [n] + [0] * (len(s) - 1)
  l, r, n = 0, 0, len(s)
  for i in range(1, n):

      l = r = i

      z[i] = r - l
      r -= 1
    else:
      k = i-l
      if z[k] < r - i + 1: z[i] = z[k]
      else:
        l = i
        while r < n and s[r-l] == s[r]: r += 1
        z[i] = r - l
        r -= 1
  return z
