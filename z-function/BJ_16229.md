def Z(s):
  z = [0]*len(s); l, r, n = 0, 0, len(s); z[0] = n
  for i in range(1, n):
    if i > r:
      l = r = i
      while r < n and s[r-l] == s[r]: r+= 1
      z[i] = r-l; r-= 1
      continue
    k = i-l
    if z[k] < r-i+1:
      z[i] = z[k]
      continue
    l = i
    while r < n and s[r-l] == s[r]: r+= 1
    z[i] = r-l; r-= 1
  return z

def appendable(x):
  for i in range(0, 9876543210, x):

n, k = map(int, input().split())
